3//! PubSub messages for follows
4use crate::{pubsub, types};
5use serde_derive::{Deserialize, Serialize};
6
7/// A user follows the channel
8#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq, Hash)]
9#[serde(into = "String", try_from = "String")]
10pub struct Following {
11    /// The channel_id to watch. Can be fetched with the [Get Users](crate::helix::users::get_users) endpoint
12    pub channel_id: u32,
13}
14
15impl_de_ser!(
16    Following,
17    "following",
18    channel_id // FIXME: add trailing comma
19);
20
21impl pubsub::Topic for Following {
22    #[cfg(feature = "twitch_oauth2")]
23    const SCOPE: twitch_oauth2::Validator = twitch_oauth2::validator![];
24
25    fn into_topic(self) -> pubsub::Topics { super::Topics::Following(self) }
26}

28/// Reply from [Following]
29#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
30#[cfg_attr(feature = "deny_unknown_fields", serde(deny_unknown_fields))]
31#[serde(tag = "type")]
32#[non_exhaustive]
33pub struct FollowingReply {
34    /// Display name of user that followed
35    pub display_name: types::DisplayName,
36    /// ID of the user that followed
37    pub user_id: types::UserId,
38    /// Name of the user that followed
39    pub username: types::UserName,
40}
