28/// A redemption users can "buy" with channel points to trigger rewards
29#[derive(Clone, Debug, PartialEq, Eq, Serialize, Deserialize)]
30#[cfg_attr(feature = "deny_unknown_fields", serde(deny_unknown_fields))]
31#[non_exhaustive]
32pub struct Redemption {
33    /// ID of channel where the redemption was triggered
34    pub channel_id: types::UserId,
35    /// ID of the redemption. Also returned in twitch IRC
36    pub id: types::RedemptionId,
37    /// Timestamp in which a reward was redeemed
38    pub redeemed_at: types::Timestamp,
39    /// Data about the reward that was redeemed
40    pub reward: Reward,
41    /// reward redemption status, will be FULFILLED if a user skips the reward queue, UNFULFILLED otherwise. ACTION_TAKEN is for `redemption-status-update`
42    pub status: RedemptionStatus,
43    /// User that triggered the reward
44    pub user: types::User,
45    /// A string that the user entered if the reward requires input
46    pub user_input: Option<String>,
47    /// A cursor for something
48    pub cursor: Option<String>,
49}
50
51/// Status for redemption
52///
53/// # Note
54///
55/// It is currently not possible to see if a unfullfilled redemption has been fullfilled.
56/// Currently only [`ACTION_TAKEN`](RedemptionStatus::ActionTaken) is given on queue changes
57#[derive(Clone, Debug, PartialEq, Eq, Serialize, Deserialize)]
58#[serde(rename_all = "UPPERCASE")]
59#[non_exhaustive]
60pub enum RedemptionStatus {
61    /// Redemption was fullfilled, e.g it skipped reward queue
62    Fulfilled,
63    /// Redemption is not fullfilled, e.g is in reward queue.
64    Unfulfilled,
65    // FIXME: https://github.com/twitchdev/issues/issues/111
66    /// Redemption was updated. Rejected or completed
67    #[serde(rename = "ACTION_TAKEN")]
68    ActionTaken,
69}

71/// Reward data
72#[derive(Clone, Debug, PartialEq, Eq, Serialize, Deserialize)]
73#[cfg_attr(feature = "deny_unknown_fields", serde(deny_unknown_fields))]
74#[non_exhaustive]
75pub struct Reward {
76    /// Color of background in rewards & challenges screen on client
77    pub background_color: String,
78    /// ID of channel where the redemption was triggered
79    pub channel_id: types::UserId,
80    /// Cooldown will expire after this timestamp
81    pub cooldown_expires_at: Option<types::Timestamp>,
82    /// Cost of reward.
83    pub cost: u32,
84    /// Default image of reward in rewards & challenges screen on client
85    pub default_image: Option<types::Image>,
86    /// Information about global cooldown
87    pub global_cooldown: types::GlobalCooldown,
88    /// ID of reward.
89    pub id: types::RewardId,
90    /// Set image of reward in rewards & challenges screen on client. If none, see [`Reward::default_image`]
91    pub image: Option<types::Image>,
92    /// Reward is enabled or not.
93    pub is_enabled: bool,
94    /// Reward is in stock
95    pub is_in_stock: bool,
96    /// Reward is paused
97    pub is_paused: bool,
98    /// Reward is sub only
99    pub is_sub_only: bool,
100    /// Reward requires input from user on rewards & challenges screen on client
101    pub is_user_input_required: bool,
102    /// Maximum redemptions per stream
103    pub max_per_stream: types::Max,
104    /// Maximum redemptions per user per stream
105    pub max_per_user_per_stream: types::Max,
106    /// Prompt shown when clicking reward on rewards & challenges screen on client
107    pub prompt: String,
108    // TODO: Is this fullfilled redemptions or is it x + 1 ? where 1 is this reward redemption
109    /// Amount of times this has been redeemed this stream.
110    pub redemptions_redeemed_current_stream: Option<u32>,
111    /// Does redemption skip queue?
112    pub should_redemptions_skip_request_queue: bool,
113    /// Template ID
114    pub template_id: Option<String>,
115    /// Title or Name of reward
116    pub title: String,
117    /// Unknown
118    pub updated_for_indicator_at: Option<types::Timestamp>,
119}

121/// `update-redemption-statuses-finished``progress
122#[derive(Clone, Debug, PartialEq, Eq, Serialize, Deserialize)]
123#[cfg_attr(feature = "deny_unknown_fields", serde(deny_unknown_fields))]
124#[non_exhaustive]
125pub struct Progress {
126    /// ID of channel user
127    pub channel_id: types::UserId,
128    /// BASE64 representation of reward
129    pub id: String,
130    /// Method by which redemptions were set to new status
131    // FIXME: BY_CHANNEL etc, need to enumify
132    pub method: String,
133    /// New status of redemptions
134    pub new_status: RedemptionStatus,
135    /// Total amount of redemptions changed
136    pub processed: i64,
137    /// Total redemptions
138    // FIXME: What is this?
139    pub total: i64,
140}

142/// Reply from [ChannelPointsChannelV1]
143#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
144#[serde(tag = "type", content = "data")]
145#[cfg_attr(feature = "deny_unknown_fields", serde(deny_unknown_fields))]
146#[non_exhaustive]
147pub enum ChannelPointsChannelV1Reply {
148    /// A reward was redeemed
149    #[serde(rename = "reward-redeemed")]
150    RewardRedeemed {
151        /// Time the pubsub message was sent
152        timestamp: String,
153        /// Data about the redemption, includes unique id and user that redeemed it
154        redemption: Redemption,
155    },
156    /// A custom reward was updated
157    #[serde(rename = "custom-reward-updated")]
158    CustomRewardUpdated {
159        /// Time the pubsub message was sent
160        timestamp: String,
161        /// Data about the reward that was updated
162        updated_reward: Reward,
163    },
164    /// Status of a redemption was changed
165    #[serde(rename = "redemption-status-update")]
166    RedemptionStatusUpdate {
167        /// Time the pubsub message was sent
168        timestamp: String,
169        /// Data about the reward that had status updated
170        redemption: Redemption,
171    },
172    /// Status of multiple redemptions were changed
173    // TODO: This seems to only be on complete all / reject all
174    #[serde(rename = "update-redemption-statuses-finished")]
175    UpdateRedemptionStatusesFinished {
176        /// Time the pubsub message was sent
177        timestamp: String,
178        /// Data about the reward that had status updated
179        progress: Progress,
180    },
181    /// Status of multiple redemptions were changed
182    // FIXME: No idea what this is, might be same as UpdateRedemptionStatusesFinished
183    #[serde(rename = "update-redemption-statuses-progress")]
184    UpdateRedemptionStatusProgress {
185        /// Time the pubsub message was sent
186        timestamp: String,
187        /// Data about the reward that had status updated
188        progress: Progress,
189    },
190}
