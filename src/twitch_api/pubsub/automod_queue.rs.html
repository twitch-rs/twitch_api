1//! PubSub messages for when AutoMod flags a message as potentially inappropriate, and when a moderator takes action on a message.
2use crate::{pubsub, types};
3use serde_derive::{Deserialize, Serialize};

5/// A user follows the channel
6#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq, Hash)]
7#[serde(into = "String", try_from = "String")]
8pub struct AutoModQueue {
9    /// The currently authenticated moderator
10    pub moderator_id: u32,
11    /// The channel_id to watch. Can be fetched with the [Get Users](crate::helix::users::get_users) endpoint
12    pub channel_id: u32,
13}
14
15impl_de_ser!(
16    AutoModQueue,
17    "automod-queue",
18    moderator_id,
19    channel_id // FIXME: add trailing comma
20);

22impl pubsub::Topic for AutoModQueue {
23    #[cfg(feature = "twitch_oauth2")]
24    const SCOPE: twitch_oauth2::Validator =
25        twitch_oauth2::validator![twitch_oauth2::Scope::ChannelModerate];
26
27    fn into_topic(self) -> pubsub::Topics { super::Topics::AutoModQueue(self) }
28}

30/// Reply from [AutoModQueue]
31#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
32#[cfg_attr(feature = "deny_unknown_fields", serde(deny_unknown_fields))]

35pub enum AutoModQueueReply {

41/// Message held by automod
42#[derive(Clone, Debug, PartialEq, Eq, Serialize, Deserialize)]
43#[cfg_attr(feature = "deny_unknown_fields", serde(deny_unknown_fields))]
44#[non_exhaustive]
45pub struct AutoModCaughtMessage {
46    /// Classification of caught message
47    pub content_classification: ContentClassification,
48    /// The message that was sent
49    pub message: Message,
50    // TODO: What is this?
51    /// Code for reason
52    #[serde(
53        default,
54        deserialize_with = "pubsub::deserialize_none_from_empty_string"
55    )]
56    pub reason_code: Option<String>,
57    /// User ID of who resolved the message in the queue
58    #[serde(
59        default,
60        deserialize_with = "pubsub::deserialize_none_from_empty_string"
61    )]
62    pub resolver_id: Option<types::UserId>,
63    /// Username of who resolved the message in the queue
64    #[serde(
65        default,
66        deserialize_with = "pubsub::deserialize_none_from_empty_string"
67    )]
68    pub resolver_login: Option<types::UserName>,
69    /// Status of the message in the queue

73/// Classification for content according to AutoMod
74#[derive(Clone, Debug, PartialEq, Eq, Serialize, Deserialize)]
75#[cfg_attr(feature = "deny_unknown_fields", serde(deny_unknown_fields))]
76#[non_exhaustive]
77pub struct ContentClassification {
78    // FIXME: Enum?
79    /// Category for classification
80    ///
81    /// On twitch, these are the different categories available for AutoMod
82    ///
83    /// * Aggression
84    ///   Threatening, inciting, or promoting violence or other harm
85    /// * Bullying: namecalling
86    ///   Name-calling, insults, or antagonization
87    /// * Disability
88    ///   Demonstrating hatred or prejudice based on perceived or actual mental or physical abilities
89    /// * Sexuality, sex, or gender
90    ///   Demonstrating hatred or prejudice based on sexual identity, sexual orientation, gender identity, or gender expression
91    /// * Misogyny: misogyny
92    ///   Demonstrating hatred or prejudice against women, including sexual objectification
93    /// * Race, ethnicity, or religion: racism
94    ///   Demonstrating hatred or prejudice based on race, ethnicity, or religion
95    /// * Sex-based terms: sexwords
96    ///   Sexual acts, anatomy
97    /// * Swearing: swearing
98    ///   Swear words, &*^!#@%
99    pub category: String,
100    /// Level of classification, eg. how strongly related the classification is related according to AutoMod
101    pub level: i64,
102}

104/// Message that was caught by AutoMod
105#[derive(Clone, Debug, PartialEq, Eq, Serialize, Deserialize)]
106#[cfg_attr(feature = "deny_unknown_fields", serde(deny_unknown_fields))]
107#[non_exhaustive]

121/// A user according to Automod
122#[derive(Clone, Debug, PartialEq, Eq, Serialize, Deserialize)]
123#[cfg_attr(feature = "deny_unknown_fields", serde(deny_unknown_fields))]
124#[non_exhaustive]
125pub struct MessageUser {

153/// The contents of a AutoMod message
154#[derive(Clone, Debug, PartialEq, Eq, Serialize, Deserialize)]
155#[cfg_attr(feature = "deny_unknown_fields", serde(deny_unknown_fields))]
156#[non_exhaustive]
157pub struct Content {
158    /// The message split up in fragments.
159    ///
160    /// The message can be retrieved in full with [`text`](Self::text)
161    pub fragments: Vec<Fragment>,
162    /// The full message that was sent
163    pub text: String,
164}
165
166/// A fragment of a AutoModded message
167///
168/// Can either be regular text, or classified as part of the reason for AutoMod
169#[derive(Clone, Debug, PartialEq, Eq, Serialize, Deserialize)]
170#[cfg_attr(feature = "deny_unknown_fields", serde(deny_unknown_fields))]
171#[serde(untagged)]
172#[non_exhaustive]
173pub enum Fragment {
174    /// Fragment that is classified under a AutoMod category which is being filtered out
175    AutomodFragment {
176        /// Text associated with this fragment
177        text: String,
178        /// AutoMod classification of the fragment
179        automod: Automod,
180    },
181    /// Fragment that is not classified under a AutoMod category
182    TextFragment {
183        /// Text associated with this fragment
184        text: String,
185    },
186    /// A text fragment that mentions another user
187    UserMention {
188        /// Text associated with this fragment
189        text: String,
190        /// User mentioned
191        user_mention: FragmentUserMention,
192    },
193}
194
195/// A mentioned user in a fragment
196#[derive(Clone, Debug, PartialEq, Eq, Serialize, Deserialize)]
197#[cfg_attr(feature = "deny_unknown_fields", serde(deny_unknown_fields))]
198#[non_exhaustive]
199pub struct FragmentUserMention {
200    /// User ID of the user
201    #[serde(rename = "userID")]
202    pub user_id: types::UserId,
203    /// Username of the user
204    pub login: types::UserName,
205    /// Display name of the user
206    pub display_name: types::DisplayName,
207}

212#[non_exhaustive]
213pub struct Automod {

225}
