3//! PubSub messages for sub gifts
4use crate::{pubsub, types};
5use serde_derive::{Deserialize, Serialize};

7/// A user gifts subs.
8///
9/// This allows one to know how many subs were gifted in a single event. See also [`pubsub::channel_subscriptions::ChannelSubscribeEventsV1`] which needs token from broadcaster
10#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq, Hash)]
11#[serde(into = "String", try_from = "String")]
12pub struct ChannelSubGiftsV1 {
13    /// The channel_id to watch. Can be fetched with the [Get Users](crate::helix::users::get_users) endpoint
14    pub channel_id: u32,
15}
16
17impl_de_ser!(
18    ChannelSubGiftsV1,
19    "channel-sub-gifts-v1",
20    channel_id // FIXME: add trailing comma
21);
22
23impl pubsub::Topic for ChannelSubGiftsV1 {
24    #[cfg(feature = "twitch_oauth2")]
25    const SCOPE: twitch_oauth2::Validator = twitch_oauth2::validator![];
26
27    fn into_topic(self) -> pubsub::Topics { super::Topics::ChannelSubGiftsV1(self) }
28}

30/// Mystery gift
31#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
32#[cfg_attr(feature = "deny_unknown_fields", serde(deny_unknown_fields))]
33#[non_exhaustive]
34pub struct MysteryGiftPurchase {
35    /// Channel where mystery gifts were distributed
36    pub channel_id: types::UserId,
37    /// Amount of mystery gifts
38    pub count: i64,
39    /// Tier of mystery gifts
40    pub tier: types::SubscriptionTier,
41    /// User ID of user that purchased mystery gifts
42    pub user_id: types::UserId,
43    /// User name of user that purchased mystery gifts
44    pub user_name: types::UserName,
45    /// Display name of user that purchased mystery gifts
46    pub display_name: types::DisplayName,
47    /// Unknown
48    pub uuid: String,
49}
50
51/// Reply from [ChannelSubGiftsV1]
52#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
53#[cfg_attr(feature = "deny_unknown_fields", serde(deny_unknown_fields))]
54#[serde(tag = "type")]
55#[non_exhaustive]
56pub enum ChannelSubGiftsV1Reply {
57    /// Mystery gift

59    MysteryGiftPurchase(MysteryGiftPurchase),

85        let actual = dbg!(Response::parse(&source).unwrap());
