1//! PubSub messages for user moderation notifications
2use crate::{pubsub, types};
3use serde_derive::{Deserialize, Serialize};
4
5/// A user follows the channel
6#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq, Hash)]
7#[serde(into = "String", try_from = "String")]
8pub struct UserModerationNotifications {
9    /// The currently authenticated user for whose automod messages will be reported on
10    pub current_user_id: u32,
11    /// The channel_id to watch. Can be fetched with the [Get Users](crate::helix::users::get_users) endpoint
12    pub channel_id: u32,
13}
14
15impl_de_ser!(
16    UserModerationNotifications,
17    "user-moderation-notifications",
18    current_user_id,
19    channel_id // FIXME: add trailing comma
20);
21
22impl pubsub::Topic for UserModerationNotifications {
23    #[cfg(feature = "twitch_oauth2")]
24    const SCOPE: twitch_oauth2::Validator = twitch_oauth2::validator![];
25
26    fn into_topic(self) -> pubsub::Topics { super::Topics::UserModerationNotifications(self) }
27}
28
29/// Reply from [UserModerationNotifications]
30#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
31#[cfg_attr(feature = "deny_unknown_fields", serde(deny_unknown_fields))]
32#[serde(tag = "type", content = "data")]
33#[non_exhaustive]
34pub enum UserModerationNotificationsReply {
35    /// Message held by automod
36    #[serde(rename = "automod_caught_message")]
37    AutoModCaught {
38        // FIXME: twitch says message.id, it's message.message_id
39        /// Identifier of the message
40        message_id: types::MsgId,
41        /// Current status of the message
42        status: types::AutomodStatus,
43    },
44}

49    #[test]
50    fn automodcaught() {
51        let source = r#"
52{
53    "type": "MESSAGE",
54    "data": {
55        "topic": "user-moderation-notifications.27620241.268131879",
56        "message": "{\"type\":\"automod_caught_message\",\"data\":{\"message_id\":\"d6f608f8-8f34-4f65-947c-0a92e31b0bfc\",\"status\":\"PENDING\"}}"
57    }
58}"#;
59        let actual = dbg!(Response::parse(source).unwrap());
60        assert!(matches!(

62            Response::Message {
63                data: TopicData::UserModerationNotifications { .. },

65        ));
66    }
