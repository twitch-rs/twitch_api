1//! Holds serializable EventSub stuff
2//!
3//! Use [`CreateEventSubSubscription`](crate::helix::eventsub::CreateEventSubSubscriptionRequest) to subscribe to an event according to the [EventSub guide](https://dev.twitch.tv/docs/eventsub).
4//! Parse the response payload text with [`Event::parse_http`] or [`Event::parse_websocket`]
5//!
6//! # Examples
7//!
8//! See [`examples/`](https://github.com/twitch-rs/twitch_api/tree/main/examples/eventsub) for a more complete example of using eventsub.

10//!
11//! Subscribe to a channel's follow events:
12//!
13//! ```rust, no_run
14//! use twitch_api::{
15//!     eventsub::{channel::ChannelFollowV2, Transport, TransportMethod},
16//!     helix,
17//! };
18//! # use twitch_api::client;
19//! # #[tokio::main]
20//! # async fn main() -> Result<(), Box<dyn std::error::Error + Send + Sync + 'static>> {
21//! # let client: helix::HelixClient<'static, client::DummyHttpClient> = helix::HelixClient::default();
22//! # let token = twitch_oauth2::AccessToken::new("validtoken".to_string());
23//! # let token = twitch_oauth2::UserToken::from_existing(&client, token, None, None).await?;
24//!
25//! let event = ChannelFollowV2::new("1234", "5678");
26//! let transport = Transport::webhook(
27//!     "https://example.org/eventsub/channelfollow",
28//!     String::from("secretabcd"),
29//! );
30//!
31//! let event_information = client
32//!     .create_eventsub_subscription(event, transport, &token)
33//!     .await?;
34//!
35//! println!("event id: {:?}", event_information.id);
36//! # Ok(())
37//! # }
38//! ```
39//!
40//! You'll now get a http POST request to the url you specified as the `callback`.
41//! You need to respond to this request from your webserver with a 200 OK response with the [`challenge`](VerificationRequest::challenge) as the body.
42//! After this, you'll get notifications
43//! ```rust
44//! use twitch_api::eventsub::{Event, Payload, Message};
45//! pub fn parse_request(
46//!     request: &http::Request<Vec<u8>>,
47//! ) -> Result<http::Response<Vec<u8>>, Box<dyn std::error::Error + Send + Sync + 'static>> {
48//!     // First, we verify the response, assuring it's legit.
49//!     if !Event::verify_payload(request, b"secretabcd") {
50//!         return Err(todo!());
51//!     }
52//!     match Event::parse_http(request)? {
53//!         Event::ChannelFollowV2(Payload {
54//!             message: Message::VerificationRequest(ver),
55//!             ..
56//!         }) => {
57//!             // We've verified the request, so we can respond to it with the challenge
58//!             Ok(http::Response::builder()
59//!                 .status(200)
60//!                 .body(ver.challenge.into_bytes())?)
61//!         },
62//!         Event::ChannelFollowV2(Payload {
63//!             message: Message::Notification(notif),
64//!             ..
65//!         }) => {
66//!             // make sure you save the `Twitch-Eventsub-Message-Id` headers value,
67//!             // twitch may resend notifications, and in those cases you should just return 200 OK.
68//!
69//!             // Do whatever you need to do with the event. Preferably send the event to a channel.
70//!             println!("user {:?} followed {:?}", notif.user_name, notif.broadcaster_user_name);
71//!             Ok(http::Response::builder().status(200).body(vec![])?)
72//!         }
73//!         _ => Ok(http::Response::builder().status(200).body(vec![])?),
74//!     }
75//! }
76//! ```

468    /// Parse http post request as a [Payload] with a specific [event](EventSubscription).
469    ///
470    /// If you don't know what event this payload is, use [`Event::parse_http`] instead.
471    ///
472    /// If your [`Request<B>`](http::Request) is of another type that doesn't implement `AsRef<[u8]>`, try converting it with [`Request::map`](http::Request::map)
473    ///
474    /// ```rust
475    /// use http::Request;
476    /// use twitch_api::eventsub::{Payload, channel::ChannelFollowV2};
477    /// # struct Body {} impl Body { fn new() -> Self {Body {}} fn to_bytes(&self) -> &[u8] { &[] } }
478    /// # fn a() -> Result<(), twitch_api::eventsub::PayloadParseError> {
479    /// // Example of a request with a body that doesn't implement `AsRef<[u8]>`
480    /// let original_request: Request<Body> = http::Request::new(Body::new());
481    /// // Convert to a request with a body of `Vec<u8>`, which does implement `AsRef<[u8]>`
482    /// let converted_request: Request<Vec<u8>> = original_request.map(|r| r.to_bytes().to_owned());
483    /// Payload::<ChannelFollowV2>::parse_http(&converted_request)?
484    /// # ; Ok(())}
485    /// ```
486    pub fn parse_http<B>(request: &http::Request<B>) -> Result<Self, PayloadParseError>
487    where B: AsRef<[u8]> {
488        // FIXME: Add some debug assertions for version and type
489
490        let source = request.body().as_ref().into();
491        let ty = request
492            .headers()
493            .get("Twitch-Eventsub-Message-Type")
494            .map(|v| v.as_bytes())
495            .unwrap_or_else(|| b"notification")
496            .into();
497        Self::parse_request(ty, source)
498    }

500    /// Parse a slice as a [`Payload`] with a specific message type. You should not use this, instead, use [`Payload::parse_http`] or the specific `parse_*` functions
501    #[doc(hidden)]
502    pub fn parse_request<'a>(
503        ty: Cow<'a, [u8]>,
504        source: Cow<'a, [u8]>,
505    ) -> Result<Self, PayloadParseError> {
506        let source = std::str::from_utf8(&source)?;
507        Self::parse_request_str(ty.as_ref(), source)
508    }
509
510    /// Parse a string slice as a [`Payload`] with a specific message type. You should not use this, instead, use [`Payload::parse_http`] or the specific `parse_*` functions
511    #[doc(hidden)]
512    pub fn parse_request_str<'a>(ty: &'a [u8], source: &'a str) -> Result<Self, PayloadParseError> {
513        match ty {
514            b"notification" => Self::parse_notification(source),
515            b"webhook_callback_verification" => Self::parse_verification_request(source),
516            b"revocation" => Self::parse_revocation(source),
517            typ => Err(PayloadParseError::UnknownMessageType(
518                String::from_utf8_lossy(typ).into_owned(),
519            )),
520        }
521    }
