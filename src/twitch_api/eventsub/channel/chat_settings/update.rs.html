90fn parse_payload() {
91    use crate::eventsub::{Event, Message};
92
93    let payload = r##"
94    {
95        "subscription": {
96            "id": "f1c2a387-161a-49f9-a165-0f21d7a4e1c4",
97            "type": "channel.chat_settings.update",
98            "version": "1",
99            "status": "enabled",
100            "cost": 0,
101            "condition": {
102                "broadcaster_user_id": "1337",
103                "user_id": "9001"
104            },
105            "transport": {
106                "method": "webhook",
107                "callback": "https://example.com/webhooks/callback"
108            },
109            "created_at": "2023-04-11T10:11:12.123Z"
110        },
111        "event": {
112            "broadcaster_user_id": "1337",
113            "broadcaster_user_login": "cool_user",
114            "broadcaster_user_name": "Cool_User",
115            "emote_mode": true,
116            "follower_mode": false,
117            "follower_mode_duration_minutes": null,
118            "slow_mode": true,
119            "slow_mode_wait_time_seconds": 10,
120            "subscriber_mode": false,
121            "unique_chat_mode": false
122        }
123    }
124    "##;
125
126    let val = Event::parse(payload).unwrap();
127    crate::tests::roundtrip(&val);
128
129    let Event::ChannelChatSettingsUpdateV1(val) = val else {
130        panic!("invalid event type");
131    };
132    let Message::Notification(notif) = val.message else {
133        panic!("invalid settings type");
134    };
135
136    assert_eq!(notif.broadcaster_user_id.as_str(), "1337");
137    assert!(notif.emote_mode);
138    assert!(!notif.follower_mode);
139    assert!(notif.follower_mode_duration_minutes.is_none());
140    assert!(notif.slow_mode);
141    assert_eq!(notif.slow_mode_wait_time_seconds, Some(10));
142    assert!(!notif.subscriber_mode);
143    assert!(!notif.unique_chat_mode);
144}
