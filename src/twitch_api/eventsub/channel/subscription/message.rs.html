2//! A user sends a resubscription chat message in a specific channel
3
4use super::*;
5/// [`channel.subscription.message`](https://dev.twitch.tv/docs/eventsub/eventsub-subscription-types#channelsubscriptionmessage): a subscription to the specified channel expires.
6#[derive(Clone, Debug, PartialEq, Eq, Serialize, Deserialize)]
7#[cfg_attr(feature = "typed-builder", derive(typed_builder::TypedBuilder))]
8#[cfg_attr(feature = "deny_unknown_fields", serde(deny_unknown_fields))]
9#[non_exhaustive]
10pub struct ChannelSubscriptionMessageV1 {
11    /// The broadcaster user ID for the channel you want to get resubscription chat message notifications for.
12    #[cfg_attr(feature = "typed-builder", builder(setter(into)))]
13    pub broadcaster_user_id: types::UserId,
14}
15
16impl ChannelSubscriptionMessageV1 {
17    /// The broadcaster user ID for the channel you want to get resubscription chat message notifications for.
18    pub fn broadcaster_user_id(broadcaster_user_id: impl Into<types::UserId>) -> Self {
19        Self {
20            broadcaster_user_id: broadcaster_user_id.into(),
21        }
22    }
23}
24
25impl EventSubscription for ChannelSubscriptionMessageV1 {
26    type Payload = ChannelSubscriptionMessageV1Payload;
27
28    const EVENT_TYPE: EventType = EventType::ChannelSubscriptionMessage;
29    #[cfg(feature = "twitch_oauth2")]
30    const SCOPE: twitch_oauth2::Validator =
31        twitch_oauth2::validator![twitch_oauth2::Scope::ChannelReadSubscriptions];
32    const VERSION: &'static str = "1";
33}
34
35/// [`channel.subscription.message`](ChannelSubscriptionMessageV1) response payload.
36#[derive(Clone, Debug, PartialEq, Eq, Serialize, Deserialize)]
37#[cfg_attr(feature = "deny_unknown_fields", serde(deny_unknown_fields))]
38#[non_exhaustive]
39pub struct ChannelSubscriptionMessageV1Payload {
40    /// The broadcaster user ID.
41    pub broadcaster_user_id: types::UserId,
42    /// The broadcaster login.
43    pub broadcaster_user_login: types::UserName,
44    /// The broadcaster display name.
45    pub broadcaster_user_name: types::DisplayName,
46    // FIXME: Twitch says this is `cumulative_total`, see https://github.com/twitchdev/issues/issues/415
47    /// The total number of months the user has been subscribed to the channel.
48    pub cumulative_months: i64,
49    /// The month duration of the subscription.
50    pub duration_months: i64,
51    /// An object that contains the resubscription message and emote information needed to recreate the message.
52    pub message: SubscriptionMessage,
53    /// The number of consecutive months the user’s current subscription has been active. This value is null if the user has opted out of sharing this information.
54    pub streak_months: Option<i64>,
55    /// The tier of the user’s subscription.
56    pub tier: types::SubscriptionTier,
57    /// The user ID of the user who sent a resubscription chat message.
58    pub user_id: types::UserId,
59    /// The user login of the user who sent a resubscription chat message.
60    pub user_login: types::UserName,
61    /// The user display name of the user who a resubscription chat message.
62    pub user_name: types::DisplayName,
63}
64
65/// A message attached to a [`ChannelSubscriptionMessageV1Payload`]
66#[derive(Clone, Debug, PartialEq, Eq, Serialize, Deserialize)]
67#[cfg_attr(feature = "deny_unknown_fields", serde(deny_unknown_fields))]
68#[non_exhaustive]
69pub struct SubscriptionMessage {
70    /// The text of the resubscription chat message.
71    pub text: String,
72    /// An array that includes the emote ID and start and end positions for where the emote appears in the text.
73    #[serde(deserialize_with = "crate::deserialize_default_from_null")]
74    pub emotes: Vec<types::EmoteOccurrence>,
75}
