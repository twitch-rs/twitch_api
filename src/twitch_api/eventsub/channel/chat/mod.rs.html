34/// A message
35// XXX: this struct can never be deny_unknown_fields
36#[derive(Clone, Debug, PartialEq, Eq, Serialize, Deserialize)]
37#[non_exhaustive]
38pub struct Message {
39    /// The chat message in plain text.
40    pub text: String,
41    /// Ordered list of chat message fragments.
42    pub fragments: Vec<Fragment>,
43}
44
45/// A chat message fragment
46#[derive(Clone, Debug, PartialEq, Eq, Serialize, Deserialize)]
47#[serde(tag = "type", rename_all = "snake_case")]
48#[non_exhaustive]
49pub enum Fragment {
50    /// A Cheermote.
51    Cheermote {
52        /// Message text in fragment
53        text: String,
54        /// A Cheermote.
55        cheermote: Cheermote,
56    },
57    /// A Emote.
58    Emote {
59        /// Message text in fragment
60        text: String,
61        /// A Emote.
62        emote: Emote,
63    },
64    /// A Mention.
65    Mention {
66        /// Message text in fragment
67        text: String,
68        /// A Mention.
69        mention: Mention,
70    },
71    /// A text fragment, see [`Fragment::text`].
72    Text {
73        /// Message text in fragment
74        text: String,
75    },
76}
77
78impl Fragment {
79    /// Get the text data
80    pub fn text(&self) -> &str {
81        match self {
82            Fragment::Cheermote { text, .. } => text,
83            Fragment::Emote { text, .. } => text,
84            Fragment::Mention { text, .. } => text,
85            Fragment::Text { text } => text,
86        }
87    }
88}

90/// A cheermote fragment
91#[derive(Clone, Debug, PartialEq, Eq, Serialize, Deserialize)]
92#[cfg_attr(feature = "deny_unknown_fields", serde(deny_unknown_fields))]
93#[non_exhaustive]
94pub struct Cheermote {
95    /// The name portion of the Cheermote string that you use in chat to cheer Bits.
96    ///
97    /// The full Cheermote string is the concatenation of {prefix} + {number of Bits}.
98    /// For example, if the prefix is “Cheer” and you want to cheer 100 Bits, the full Cheermote string is Cheer100.
99    /// When the Cheermote string is entered in chat, Twitch converts it to the image associated with the Bits tier that was cheered.
100    pub prefix: String,
101    /// The amount of bits cheered.
102    pub bits: i32,
103    /// The tier level of the cheermote.
104    pub tier: i32,
105}

107/// An emote fragment
108#[derive(Clone, Debug, PartialEq, Eq, Serialize, Deserialize)]
109#[cfg_attr(feature = "deny_unknown_fields", serde(deny_unknown_fields))]
110#[non_exhaustive]
111pub struct Emote {
112    /// An ID that uniquely identifies this emote.
113    pub id: types::EmoteId,
114    /// An ID that identifies the emote set that the emote belongs to.
115    pub emote_set_id: types::EmoteSetId,
116    /// The ID of the broadcaster who owns the emote.
117    pub owner_id: types::UserId,
118    /// The formats that the emote is available in. For example, if the emote is available only as a static PNG, the array contains only static. But if the emote is available as a static PNG and an animated GIF, the array contains static and animated. The possible formats are:
119    ///
120    /// * `animated` — An animated GIF is available for this emote.
121    /// * `static` — A static PNG file is available for this emote.
122    pub format: Vec<types::EmoteAnimationSetting>,
123}
