11/// A request is a Twitch endpoint, see [New Twitch API](https://dev.twitch.tv/docs/api/reference) reference
12#[async_trait::async_trait]
13pub trait Request: serde::Serialize + Sized {
14    /// The path to the endpoint relative to the helix root. eg. `channels` for [Get Channel Information](https://dev.twitch.tv/docs/api/reference#get-channel-information)
15    const PATH: &'static str;
16    /// Scopes needed for this endpoint
17    #[cfg(feature = "twitch_oauth2")]
18    const SCOPE: twitch_oauth2::Validator;
19    /// Optional scopes needed by this endpoint
20    #[cfg(feature = "twitch_oauth2")]
21    const OPT_SCOPE: &'static [twitch_oauth2::Scope] = &[];
22
23    /// Response type. twitch's response will  deserialize to this.
24    type Response: serde::de::DeserializeOwned + PartialEq;
25
26    /// Data to keep track of the pagination state.
27    ///
28    /// If the request isn't paginated, this should be `()`.
29    type PaginationData: PartialEq + Debug + PaginationData<Self>;
30
31    /// Defines layout of the url parameters.
32    fn query(&self) -> Result<String, errors::SerializeError> { ser::to_string(self) }
33    /// Returns full URI for the request, including query parameters.
34    fn get_uri(&self) -> Result<http::Uri, InvalidUri> {
35        let query = self.query()?;
36        let url = crate::TWITCH_HELIX_URL
37            .join(<Self as Request>::PATH)
38            .map(|mut u| {
39                u.set_query(Some(&query));
40                u
41            })?;
42        http::Uri::from_str(url.as_str()).map_err(Into::into)
43    }
44    /// Returns bare URI for the request, NOT including query parameters.
45    fn get_bare_uri() -> Result<http::Uri, InvalidUri> {
46        let url = crate::TWITCH_HELIX_URL.join(<Self as Request>::PATH)?;
47        http::Uri::from_str(url.as_str()).map_err(Into::into)
48    }
49}

51/// Helix endpoint POSTs information
52pub trait RequestPost: Request {
53    /// Body parameters
54    type Body: HelixRequestBody;
55
56    /// Create a [`http::Request`] from this [`Request`] in your client
57    fn create_request(
58        &self,
59        body: Self::Body,
60        token: &str,
61        client_id: &str,
62    ) -> Result<http::Request<hyper::body::Bytes>, CreateRequestError> {
63        let uri = self.get_uri()?;
64
65        let body = body.try_to_body()?;
66        // eprintln!("\n\nbody is ------------ {:?} ------------", body);
67
68        let mut bearer = http::HeaderValue::from_str(&format!("Bearer {token}")).map_err(|_| {
69            CreateRequestError::Custom("Could not make token into headervalue".into())
70        })?;
71        bearer.set_sensitive(true);
72        http::Request::builder()
73            .method(http::Method::POST)
74            .uri(uri)
75            .header("Client-ID", client_id)
76            .header("Content-Type", "application/json")
77            .header(http::header::AUTHORIZATION, bearer)
78            .body(body)
79            .map_err(Into::into)
80    }
81
82    /// Parse response.
83    ///
84    /// # Notes
85    ///
86    /// Pass in the request to enable [pagination](Response::get_next) if supported.
87    fn parse_response<B: Into<hyper::body::Bytes>>(
88        // FIXME: Is this really needed? Its currently only used for error reporting.
89        request: Option<Self>,
90        uri: &http::Uri,
91        response: http::Response<B>,
92    ) -> Result<Response<Self, <Self as Request>::Response>, HelixRequestPostError>
93    where
94        Self: Sized,
95    {
96        let response: http::Response<hyper::body::Bytes> = response.map(|b| b.into());
97        let text = std::str::from_utf8(response.body().as_ref()).map_err(|e| {
98            HelixRequestPostError::Utf8Error(response.body().clone(), e, uri.clone())
99        })?;
100        if let Ok(HelixRequestError {
101            error,
102            status,
103            message,
104        }) = parse_json::<HelixRequestError>(text, false)
105        {
106            return Err(HelixRequestPostError::Error {
107                error,
108                status: status.try_into().unwrap_or(http::StatusCode::BAD_REQUEST),
109                message,
110                uri: uri.clone(),
111                body: response.body().clone(),
112            });
113        }
114        <Self as RequestPost>::parse_inner_response(request, uri, text, response.status())
115    }
116
117    /// Parse a response string into the response.
118    fn parse_inner_response(
119        request: Option<Self>,
120        uri: &http::Uri,
121        response: &str,
122        status: http::StatusCode,
123    ) -> Result<Response<Self, <Self as Request>::Response>, HelixRequestPostError>
124    where
125        Self: Sized,
126    {
127        let response: InnerResponse<<Self as Request>::Response> = parse_json(response, true)
128            .map_err(|e| {
129                HelixRequestPostError::DeserializeError(
130                    response.to_string(),
131                    e,
132                    uri.clone(),
133                    status,
134                )
135            })?;
136        Ok(Response::new(
137            response.data,
138            Self::PaginationData::new(response.pagination.cursor, request, response.total),
139            response.other,
140        ))
141    }
142}

445/// Parses a response where Helix responds with a single datum inside `data`.
446///
447/// An example response is `{ "data": [ { "foo": 1 } ]`.
448pub(crate) fn parse_single_return<T, E>(
449    request: Option<T>,
450    uri: &http::Uri,
451    response: &str,
452    status: http::StatusCode,
453) -> Result<Response<T, T::Response>, E>
454where
455    T: Request,
456    E: errors::HelixRequestDeserError,
457{
458    let resp = match status {
459        http::StatusCode::OK => {
460            let resp: InnerResponse<[T::Response; 1]> = parse_json(response, true)
461                .map_err(|e| E::deserialize_error(response.to_string(), e, uri.clone(), status))?;
462            let [s] = resp.data;
463            s
464        }
465        _ => {
466            return Err(E::invalid_response(
467                "unexpected status code",
468                response.to_string(),
469                status,
470                uri.clone(),
471            ));
472        }
473    };
474    Ok(Response::with_data(resp, request))
475}

477/// A helper type to parse responses that contain either zero or one element in an array
478pub(crate) struct ZeroOrOne<T>(pub Option<T>);
479
480impl<'de, T> serde::Deserialize<'de> for ZeroOrOne<T>
481where T: serde::Deserialize<'de>
482{
483    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
484    where D: serde::Deserializer<'de> {
485        struct ZeroOrOneVisitor<T> {
486            _marker: PhantomData<T>,
487        }
488
489        impl<'de, T> serde::de::Visitor<'de> for ZeroOrOneVisitor<T>
490        where T: serde::Deserialize<'de>
491        {
492            type Value = ZeroOrOne<T>;
493
494            fn expecting(&self, formatter: &mut std::fmt::Formatter) -> std::fmt::Result {
495                formatter.write_str("an array of length zero or one")
496            }
497
498            fn visit_seq<A>(self, mut seq: A) -> Result<Self::Value, A::Error>
499            where A: serde::de::SeqAccess<'de> {
500                match seq.next_element() {
501                    Ok(None) => Ok(ZeroOrOne(None)),
502                    Ok(Some(it)) => match seq.next_element::<T>() {
503                        Ok(None) => Ok(ZeroOrOne(Some(it))),
504                        Ok(Some(_)) => Err(serde::de::Error::invalid_length(2, &self)),
505                        Err(e) => Err(e),
506                    },
507                    Err(e) => Err(e),
508                }
509            }
510        }
511
512        deserializer.deserialize_seq(ZeroOrOneVisitor::<T> {
513            _marker: Default::default(),
514        })
515    }
