1//! Responses contains the return values of a [request](super::Request).

4/// Response retrieved from endpoint. Data is the type in [`Request::Response`]
5#[derive(PartialEq, Eq, Debug)]
6#[non_exhaustive]
7pub struct Response<R, D>
8where
9    R: Request,
10    D: serde::de::DeserializeOwned + PartialEq, {
11    /// Twitch's response field for `data`.
12    pub data: D,
13    /// A cursor value, to be used in a subsequent request to specify the starting point of the next set of results.
14    pub pagination: Option<Cursor>,
15    /// The request that was sent, used for [pagination](super::Paginated).
16    pub request: Option<R>,
17    /// Response would return this many results if fully paginated. Sometimes this is not emmitted or correct for this purpose, in those cases, this value will be `None`.
18    pub total: Option<i64>,
19    /// Fields which are not part of the data response, but are returned by the endpoint.
20    ///
21    /// See for example [Get Broadcaster Subscriptions](https://dev.twitch.tv/docs/api/reference#get-broadcaster-subscriptions) which returns this.
22    pub other: Option<serde_json::Map<String, serde_json::Value>>,
23}

25impl<R, D> Response<R, D>
26where
27    R: Request,
28    D: serde::de::DeserializeOwned + PartialEq,
29{
30    /// Create a new response.
31    ///
32    /// # Examples
33    ///
34    /// ```rust
35    /// use twitch_api::helix::{self, Request, RequestPost};
36    ///
37    /// #[derive(serde_derive::Serialize, Debug, PartialEq)]
38    /// pub struct MyTwitchRequest {}
39    ///
40    /// #[derive(serde_derive::Deserialize, Debug, PartialEq)]
41    /// pub enum MyTwitchResponse {
42    ///     Success,
43    /// }
44    ///
45    /// impl Request for MyTwitchRequest {
46    ///     type Response = MyTwitchResponse;
47    ///
48    ///     const PATH: &'static str = "my/request";
49    ///     const SCOPE: twitch_api::twitch_oauth2::Validator =
50    ///         twitch_api::twitch_oauth2::validator![];
51    /// }
52    ///
53    /// impl RequestPost for MyTwitchRequest {
54    ///     type Body = helix::EmptyBody;
55    ///
56    ///     fn parse_inner_response(
57    ///         request: Option<Self>,
58    ///         uri: &http::Uri,
59    ///         response: &str,
60    ///         status: http::StatusCode,
61    ///     ) -> Result<
62    ///         helix::Response<Self, Self::Response>,
63    ///         helix::HelixRequestPostError,
64    ///     >
65    ///     where
66    ///         Self: Sized,
67    ///     {
68    ///         match status {
69    ///             http::StatusCode::NO_CONTENT => Ok(helix::Response::new(
70    ///                 MyTwitchResponse::Success,
71    ///                 None,
72    ///                 request,
73    ///                 None,
74    ///                 None,
75    ///             )),
76    ///             _ => Err(helix::HelixRequestPostError::InvalidResponse {
77    ///                 reason: "unexpected status",
78    ///                 response: response.to_string(),
79    ///                 status,
80    ///                 uri: uri.clone(),
81    ///             }),
82    ///         }
83    ///     }
84    /// }
85    /// ```
86    pub fn new(
87        data: D,
88        pagination: Option<Cursor>,
89        request: Option<R>,
90        total: Option<i64>,
91        other: Option<serde_json::Map<String, serde_json::Value>>,
92    ) -> Self {
93        Self {
94            data,
95            pagination,
96            request,
97            total,
98            other,
99        }
100    }
101
102    /// Create a new response without pagination, total or other fields.
103    ///
104    /// # Examples
105    ///
106    /// ```rust
107    /// use twitch_api::helix::{self, Request, RequestPut};
108    ///
109    /// #[derive(serde_derive::Serialize, Debug, PartialEq)]
110    /// pub struct MyTwitchRequest {}
111    ///
112    /// #[derive(serde_derive::Deserialize, Debug, PartialEq)]
113    /// pub enum MyTwitchResponse {
114    ///     Success,
115    /// }
116    ///
117    /// impl Request for MyTwitchRequest {
118    ///     type Response = MyTwitchResponse;
119    ///
120    ///     const PATH: &'static str = "my/request";
121    ///     const SCOPE: twitch_api::twitch_oauth2::Validator =
122    ///         twitch_api::twitch_oauth2::validator![];
123    /// }
124    ///
125    /// impl RequestPut for MyTwitchRequest {
126    ///     type Body = helix::EmptyBody;
127    ///
128    ///     fn parse_inner_response(
129    ///         request: Option<Self>,
130    ///         uri: &http::Uri,
131    ///         response: &str,
132    ///         status: http::StatusCode,
133    ///     ) -> Result<
134    ///         helix::Response<Self, Self::Response>,
135    ///         helix::HelixRequestPutError,
136    ///     >
137    ///     where
138    ///         Self: Sized,
139    ///     {
140    ///         match status {
141    ///             http::StatusCode::NO_CONTENT => Ok(helix::Response::with_data(
142    ///                 MyTwitchResponse::Success,
143    ///                 request,
144    ///             )),
145    ///             _ => Err(helix::HelixRequestPutError::InvalidResponse {
146    ///                 reason: "unexpected status",
147    ///                 response: response.to_string(),
148    ///                 status,
149    ///                 uri: uri.clone(),
150    ///             }),
151    ///         }
152    ///     }
153    /// }
154    /// ```
155    pub fn with_data(data: D, request: Option<R>) -> Self {
156        Self::new(data, None, request, None, None)
157    }
158
159    /// Get a field from the response that is not part of `data`.
160    pub fn get_other<Q, V>(&self, key: &Q) -> Result<Option<V>, serde_json::Error>
161    where
162        String: std::borrow::Borrow<Q>,
163        Q: ?Sized + Ord + Eq + std::hash::Hash,
164        V: serde::de::DeserializeOwned, {
165        use std::borrow::Borrow as _;
166        match &key {
167            total if &String::from("total").borrow() == total => {
168                if let Some(total) = self.total {
169                    let total = serde_json::json!(total);
170                    Some(serde_json::from_value(total)).transpose()
171                } else {
172                    Ok(None)
173                }
174            }
175            _ => self
176                .other
177                .as_ref()
178                .and_then(|map| map.get(key))
179                .map(|v| serde_json::from_value(v.clone()))
180                .transpose(),
181        }
182    }
183}

203#[cfg(feature = "client")]
204impl<R, D> Response<R, D>
205where
206    R: Request<Response = D> + Clone + super::Paginated + super::RequestGet + std::fmt::Debug,
207    D: serde::de::DeserializeOwned + std::fmt::Debug + PartialEq,
208{
209    /// Get the next page in the responses.
210    pub async fn get_next<'a, C: crate::HttpClient + 'a>(
211        self,
212        client: &'a super::HelixClient<'a, C>,
213        token: &(impl super::TwitchToken + ?Sized),
214    ) -> Result<Option<Response<R, D>>, super::ClientRequestError<<C as crate::HttpClient>::Error>>
215    {
216        if let Some(mut req) = self.request.clone() {
217            if self.pagination.is_some() {
218                req.set_pagination(self.pagination);
219                let res = client.req_get(req, token).await.map(Some);
220                if let Ok(Some(r)) = res {
221                    // FIXME: Workaround for https://github.com/twitchdev/issues/issues/18
222                    if r.data == self.data {
223                        Ok(None)
224                    } else {
225                        Ok(Some(r))
226                    }
227                } else {
228                    res
229                }
230            } else {
231                Ok(None)
232            }
233        } else {
234            // TODO: Make into proper error
235            Err(super::ClientRequestError::Custom(
236                "no source request attached".into(),
237            ))
238        }
239    }
