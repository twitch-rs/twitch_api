20//! use twitch_api::helix::{self, streams::get_stream_key};

39/// Query Parameters for [Get Stream Key](super::get_stream_key)
40///

42#[derive(PartialEq, Eq, Deserialize, Serialize, Clone, Debug)]
43#[cfg_attr(feature = "typed-builder", derive(typed_builder::TypedBuilder))]
44#[non_exhaustive]
45pub struct GetStreamKeyRequest<'a> {
46    /// The ID of the broadcaster that owns the channel. The ID must match the user ID in the access token.
47    #[cfg_attr(feature = "typed-builder", builder(setter(into)))]
48    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
49    pub broadcaster_id: Cow<'a, types::UserIdRef>,
50}
51
52impl<'a> GetStreamKeyRequest<'a> {
53    /// ID of the broadcaster whose stream key should be retrieved
54    pub fn broadcaster_id(broadcaster_id: impl types::IntoCow<'a, types::UserIdRef> + 'a) -> Self {
55        Self {
56            broadcaster_id: broadcaster_id.into_cow(),
57        }
58    }
59}

95#[cfg(test)]
96#[test]
97fn test_request() {
98    use helix::*;
99    let req = GetStreamKeyRequest::broadcaster_id("198704263");
100
101    // From twitch docs
102    let data = br#"
103        {
104          "data": [
105            {
106              "stream_key": "live_44322889_a34ub37c8ajv98a0"
107            }
108          ]
109        }
110    "#
111    .to_vec();
112
113    let http_response = http::Response::builder().body(data).unwrap();
114
115    let uri = req.get_uri().unwrap();
116    assert_eq!(
117        uri.to_string(),
118        "https://api.twitch.tv/helix/streams/key?broadcaster_id=198704263"
119    );
120
121    let res = GetStreamKeyRequest::parse_response(Some(req), &uri, http_response)
122        .unwrap()
123        .data;
124
125    assert_eq!(res.stream_key.as_str(), "live_44322889_a34ub37c8ajv98a0");
126}
