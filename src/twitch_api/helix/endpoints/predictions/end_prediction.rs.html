1//! End a prediction that is currently active.
2//!
3//! [`end-prediction`](https://dev.twitch.tv/docs/api/reference#end-prediction)
4//!
5//! # Accessing the endpoint
6//!
7//! ## Request: [EndPredictionRequest]
8//!
9//! To use this endpoint, construct an [`EndPredictionRequest`] with the [`EndPredictionRequest::new()`] method.
10//!
11//! ```rust
12//! use twitch_api::helix::predictions::end_prediction;
13//! let request = end_prediction::EndPredictionRequest::new();
14//! ```
15//!
16//! ## Body: [EndPredictionBody]
17//!
18//! We also need to provide a body to the request containing what we want to change.
19//!
20//! ```
21//! use twitch_api::helix::predictions::{self, end_prediction};
22//! let body = end_prediction::EndPredictionBody::new(
23//!     "141981764",
24//!     "ed961efd-8a3f-4cf5-a9d0-e616c590cd2a",
25//!     twitch_types::PredictionStatus::Resolved,
26//! );
27//! ```
28//!
29//! ## Response: [Prediction]
30//!
31//!
32//! Send the request to receive the response with [`HelixClient::req_patch()`](helix::HelixClient::req_patch).
33//!
34//!
35//! ```rust, no_run
36//! use twitch_api::helix::{self, predictions::{self, end_prediction}};
37//! # use twitch_api::client;
38//! # #[tokio::main]
39//! # async fn main() -> Result<(), Box<dyn std::error::Error + Send + Sync + 'static>> {
40//! # let client: helix::HelixClient<'static, client::DummyHttpClient> = helix::HelixClient::default();
41//! # let token = twitch_oauth2::AccessToken::new("validtoken".to_string());
42//! # let token = twitch_oauth2::UserToken::from_existing(&client, token, None, None).await?;
43//! let request = end_prediction::EndPredictionRequest::new();
44//! let body = end_prediction::EndPredictionBody::new(
45//!     "141981764",
46//!     "ed961efd-8a3f-4cf5-a9d0-e616c590cd2a",
47//!     twitch_types::PredictionStatus::Resolved
48//! );
49//! let response: predictions::Prediction = client.req_patch(request, body, &token).await?.data;
50//! # Ok(())
51//! # }
52//! ```
53//!
54//! You can also get the [`http::Request`] with [`request.create_request(&token, &client_id)`](helix::RequestPost::create_request)
55//! and parse the [`http::Response`] with [`EndPredictionRequest::parse_response(None, &request.get_uri(), response)`](EndPredictionRequest::parse_response)

78/// Body Parameters for [End Prediction](super::end_prediction)
79///
80/// [`end-prediction`](https://dev.twitch.tv/docs/api/reference#end-prediction)
81#[derive(PartialEq, Eq, Deserialize, Serialize, Clone, Debug)]
82#[cfg_attr(feature = "typed-builder", derive(typed_builder::TypedBuilder))]
83#[non_exhaustive]
84pub struct EndPredictionBody<'a> {
85    /// The broadcaster running predictions. Provided broadcaster_id must match the user_id in the user OAuth token.
86    #[cfg_attr(feature = "typed-builder", builder(setter(into)))]
87    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
88    pub broadcaster_id: Cow<'a, types::UserIdRef>,
89    /// ID of the prediction.
90    #[cfg_attr(feature = "typed-builder", builder(setter(into)))]
91    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
92    pub id: Cow<'a, types::PredictionIdRef>,
93    /// The Prediction status to be set. Valid values:
94    ///
95    /// [`RESOLVED`](types::PredictionStatus): A winning outcome has been chosen and the Channel Points have been distributed to the users who predicted the correct outcome.
96    /// [`CANCELED`](types::PredictionStatus): The Prediction has been canceled and the Channel Points have been refunded to participants.
97    /// [`LOCKED`](types::PredictionStatus): The Prediction has been locked and viewers can no longer make predictions.
98    pub status: types::PredictionStatus,
99    /// ID of the winning outcome for the Prediction. This parameter is required if status is being set to [`RESOLVED`](types::PredictionStatus).
100    #[cfg_attr(feature = "typed-builder", builder(default, setter(into)))]
101    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
102    pub winning_outcome_id: Option<Cow<'a, types::PredictionOutcomeIdRef>>,
103}

105impl<'a> EndPredictionBody<'a> {
106    /// End given prediction that is currently active.
107    pub fn new(
108        broadcaster_id: impl types::IntoCow<'a, types::UserIdRef> + 'a,
109        id: impl types::IntoCow<'a, types::PredictionIdRef> + 'a,
110        status: impl Into<types::PredictionStatus>,
111    ) -> Self {
112        Self {
113            broadcaster_id: broadcaster_id.into_cow(),
114            id: id.into_cow(),
115            status: status.into(),
116            winning_outcome_id: None,
117        }
118    }
119
120    /// ID of the winning outcome for the Prediction
121    ///
122    /// This parameter is required if status is being set to [`RESOLVED`](types::PredictionStatus).
123    pub fn winning_outcome_id(
124        mut self,
125        winning_outcome_id: impl types::IntoCow<'a, types::PredictionOutcomeIdRef> + 'a,
126    ) -> Self {
127        self.winning_outcome_id = Some(winning_outcome_id.into_cow());
128        self
129    }
130}
