1//! Blocks the specified user on behalf of the authenticated user.
2//! [`block-user`](https://dev.twitch.tv/docs/api/reference#block-user)
3//!
4//! # Accessing the endpoint
5//!
6//! ## Request: [BlockUserRequest]
7//!
8//! To use this endpoint, construct a [`BlockUserRequest`] with the [`BlockUserRequest::block_user()`] method.
9//!
10//! ```rust
11//! use twitch_api::helix::users::block_user::{self, Reason, SourceContext};
12//! let request = block_user::BlockUserRequest::block_user("1234");
13//! // Or, specifying a reason for the block
14//! let request = block_user::BlockUserRequest::block_user("1234")
15//!     .source_context(SourceContext::Chat)
16//!     .reason(Reason::Spam);
17//! ```
18//!
19//! ## Response: [BlockUser]
20//!
21//! Send the request to receive the response with [`HelixClient::req_put()`](helix::HelixClient::req_put).
22//!
23//! ```rust, no_run
24//! use twitch_api::helix::{self, users::block_user};
25//! # use twitch_api::client;
26//! # #[tokio::main]
27//! # async fn main() -> Result<(), Box<dyn std::error::Error + Send + Sync + 'static>> {
28//! # let client: helix::HelixClient<'static, client::DummyHttpClient> = helix::HelixClient::default();
29//! # let token = twitch_oauth2::AccessToken::new("validtoken".to_string());
30//! # let token = twitch_oauth2::UserToken::from_existing(&client, token, None, None).await?;
31//! let request = block_user::BlockUserRequest::block_user("1234");
32//! let response: block_user::BlockUser = client.req_put(request, helix::EmptyBody, &token).await?.data;
33//! # Ok(())
34//! # }
35//! ```
36//!
37//! You can also get the [`http::Request`] with [`request.create_request(&token, &client_id)`](helix::RequestPut::create_request)
38//! and parse the [`http::Response`] with [`BlockUserRequest::parse_response(None, &request.get_uri(), response)`](BlockUserRequest::parse_response)

43/// Query Parameters for [Block User](super::block_user)
44///
45/// [`block-user`](https://dev.twitch.tv/docs/api/reference#block-user)
46#[derive(PartialEq, Eq, Deserialize, Serialize, Clone, Debug)]
47#[cfg_attr(feature = "typed-builder", derive(typed_builder::TypedBuilder))]
48#[must_use]
49#[non_exhaustive]
50pub struct BlockUserRequest<'a> {
51    /// User ID of the follower
52    #[cfg_attr(feature = "typed-builder", builder(setter(into)))]
53    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
54    pub target_user_id: Cow<'a, types::UserIdRef>,
55    /// Source context for blocking the user. Valid values: "chat", "whisper".
56    #[cfg_attr(feature = "typed-builder", builder(default, setter(into)))]
57    pub source_context: Option<SourceContext>,
58    /// Reason for blocking the user. Valid values: "spam", "harassment", or "other".
59    #[cfg_attr(feature = "typed-builder", builder(default, setter(into)))]
60    pub reason: Option<Reason>,
61}
62
63impl<'a> BlockUserRequest<'a> {
64    /// Block a user
65    pub fn block_user(target_user_id: impl types::IntoCow<'a, types::UserIdRef> + 'a) -> Self {
66        Self {
67            target_user_id: target_user_id.into_cow(),
68            source_context: None,
69            reason: None,
70        }
71    }
72
73    /// Set the source_context for this block.
74    pub fn source_context(mut self, source_context: impl Into<SourceContext>) -> Self {
75        self.source_context = Some(source_context.into());
76        self
77    }
78
79    /// Set the reason for this block.
80    pub fn reason(mut self, reason: impl Into<Reason>) -> Self {
81        self.reason = Some(reason.into());
82        self
83    }
84}

120impl Request for BlockUserRequest<'_> {
121    type Response = BlockUser;

131impl RequestPut for BlockUserRequest<'_> {
132    type Body = helix::EmptyBody;
133
134    fn parse_inner_response(
135        request: Option<Self>,
136        uri: &http::Uri,
137        response: &str,
138        status: http::StatusCode,
139    ) -> Result<helix::Response<Self, Self::Response>, helix::HelixRequestPutError>
140    where
141        Self: Sized,
142    {
143        match status {
144            http::StatusCode::NO_CONTENT => {
145                Ok(helix::Response::with_data(BlockUser::Success, request))
146            }
147            _ => Err(helix::HelixRequestPutError::InvalidResponse {
148                reason: "unexpected status",
149                response: response.to_string(),
150                status,
151                uri: uri.clone(),
152            }),
153        }
154    }
155}
