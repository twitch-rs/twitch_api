1//! Updates the specified user’s information.
2//!
3//! [`update-user`](https://dev.twitch.tv/docs/api/reference/#update-user)
4//!
5//! The user ID in the OAuth token identifies the user whose information you want to update.
6//!
7//! # Accessing the endpoint
8//!
9//! ## Request: [UpdateUserRequest]
10//!
11//! To use this endpoint, construct an [`UpdateUserRequest`] with the [`UpdateUserRequest::description()`] method.
12//!
13//! ```rust
14//! use twitch_api::helix::users::update_user;
15//! let mut request =
16//!     update_user::UpdateUserRequest::description("my description");
17//! ```
18//!
19//! ## Response: [User]
20//!
21//!
22//! Send the request to receive the response with [`HelixClient::req_patch()`](helix::HelixClient::req_patch).
23//!
24//!
25//! ```rust, no_run
26//! use twitch_api::helix::{self, users::update_user};
27//! # use twitch_api::client;
28//! # #[tokio::main]
29//! # async fn main() -> Result<(), Box<dyn std::error::Error + Send + Sync + 'static>> {
30//! # let client: helix::HelixClient<'static, client::DummyHttpClient> = helix::HelixClient::default();
31//! # let token = twitch_oauth2::AccessToken::new("validtoken".to_string());
32//! # let token = twitch_oauth2::UserToken::from_existing(&client, token, None, None).await?;
33//! let mut request = update_user::UpdateUserRequest::description("my description");
34//! let body = helix::EmptyBody;
35//! let response: helix::users::User = client.req_put(request, body, &token).await?.data;
36//! # Ok(())
37//! # }
38//! ```
39//!
40//! You can also get the [`http::Request`] with [`request.create_request(&token, &client_id)`](helix::RequestPost::create_request)
41//! and parse the [`http::Response`] with [`UpdateUserRequest::parse_response(None, &request.get_uri(), response)`](UpdateUserRequest::parse_response)

45/// Query Parameters for [Update User](super::update_user)
46///
47/// [`update-user`](https://dev.twitch.tv/docs/api/reference#update-user)
48#[derive(PartialEq, Eq, Deserialize, Serialize, Clone, Debug)]
49#[cfg_attr(feature = "typed-builder", derive(typed_builder::TypedBuilder))]
50#[must_use]
51#[non_exhaustive]
52pub struct UpdateUserRequest<'a> {
53    /// The string to update the channel’s description to. The description is limited to a maximum of 300 characters.
54    ///
55    /// To remove the description, specify this parameter but don’t set it’s value (specify an empty string).
56    #[cfg_attr(feature = "typed-builder", builder(default, setter(into)))]
57    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
58    pub description: Option<Cow<'a, str>>,
59}
60
61impl<'a> UpdateUserRequest<'a> {
62    /// Update nothing (returns the current user)
63    pub fn empty() -> Self { Self { description: None } }
64
65    /// Update the description of the current user
66    pub fn description(description: impl Into<Cow<'a, str>>) -> Self {
67        Self {
68            description: Some(description.into()),
69        }
70    }
71}
72
73impl Request for UpdateUserRequest<'_> {
74    type Response = User;
75
76    const PATH: &'static str = "users";
77    #[cfg(feature = "twitch_oauth2")]
78    const SCOPE: twitch_oauth2::Validator =
79        twitch_oauth2::validator![twitch_oauth2::Scope::UserEdit];
80}

82impl RequestPut for UpdateUserRequest<'_> {

94        helix::parse_single_return(request, uri, response, status)
95    }
96}

98#[cfg(test)]
99#[test]
100fn test_request() {
101    let req = UpdateUserRequest::description("my description");
102
103    let body = helix::EmptyBody;
104
105    dbg!(req.create_request(body, "token", "clientid").unwrap());
106
107    // From twitch docs (slightly modified to include a space)
108    let data = br#"
109        {
110          "data":[{
111            "id": "44322889",
112            "login": "dallas",
113            "display_name": "dallas",
114            "type": "staff",
115            "broadcaster_type": "affiliate",
116            "description": "my description",
117            "profile_image_url": "https://static-cdn.jtvnw.net/jtv_user_pictures/4d1f36cbf1f0072d-profile_image-300x300.png",
118            "offline_image_url": "https://static-cdn.jtvnw.net/jtv_user_pictures/dallas-channel_offline_image-2e82c1df2a464df7-1920x1080.jpeg",
119            "view_count": 6995,
120            "email": "not-real@email.com",
121            "created_at": "2013-06-03T19:12:02.580593Z"
122          }]
123        }
124    "#.to_vec();
125
126    let http_response = http::Response::builder().status(200).body(data).unwrap();
127
128    let uri = req.get_uri().unwrap();
129    assert_eq!(
130        uri.to_string(),
131        "https://api.twitch.tv/helix/users?description=my+description"
132    );
133
134    let res = UpdateUserRequest::parse_response(Some(req), &uri, http_response)
135        .unwrap()
136        .data;
137
138    assert_eq!(res.id.as_str(), "44322889");
139    assert_eq!(res.type_, Some(types::UserType::Staff));
140    assert_eq!(
141        res.broadcaster_type,
142        Some(types::BroadcasterType::Affiliate)
143    );
144    assert_eq!(res.login.as_str(), "dallas");
145    assert_eq!(res.description.unwrap(), "my description");
146}
