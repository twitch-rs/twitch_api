13//! ```

17//!

92impl RequestPost for StartARaidRequest<'_> {
93    type Body = helix::EmptyBody;
94
95    fn parse_inner_response(
96        request: Option<Self>,
97        uri: &http::Uri,
98        response_str: &str,
99        status: http::StatusCode,
100    ) -> Result<helix::Response<Self, Self::Response>, helix::HelixRequestPostError>
101    where
102        Self: Sized,
103    {
104        let response: helix::InnerResponse<Vec<Self::Response>> =
105            helix::parse_json(response_str, true).map_err(|e| {
106                helix::HelixRequestPostError::DeserializeError(
107                    response_str.to_string(),
108                    e,
109                    uri.clone(),
110                    status,
111                )
112            })?;
113        let data = response.data.into_iter().next().ok_or_else(|| {
114            helix::HelixRequestPostError::InvalidResponse {
115                reason: "response included no data",
116                response: response_str.to_string(),
117                status,
118                uri: uri.clone(),
119            }
120        })?;
121        Ok(helix::Response {
122            data,
123            pagination: response.pagination.cursor,
124            request,
125            total: None,
126            other: None,
127        })
128    }
129}
