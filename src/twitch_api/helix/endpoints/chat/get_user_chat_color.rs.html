2//! [`get-user-chat-color`](https://dev.twitch.tv/docs/api/reference#get-user-chat-color)
3//!
4//! # Accessing the endpoint
5//!
6//! ## Request: [GetUserChatColorRequest]
7//!
8//! To use this endpoint, construct a [`GetUserChatColorRequest`] with the [`GetUserChatColorRequest::user_ids()`] method.
9//!
10//! ```rust
11//! use twitch_api::helix::chat::get_user_chat_color;
12//! let request =
13//!     get_user_chat_color::GetUserChatColorRequest::user_ids(&["4321"]);
14//! ```
15//!
16//! ## Response: [UserChatColor]
17//!
18//! Send the request to receive the response with [`HelixClient::req_get()`](helix::HelixClient::req_get).
19//!
20//! ```rust, no_run
21//! use twitch_api::helix::{self, chat::get_user_chat_color};

23//! # #[tokio::main]
24//! # async fn main() -> Result<(), Box<dyn std::error::Error + Send + Sync + 'static>> {
25//! # let client: helix::HelixClient<'static, client::DummyHttpClient> = helix::HelixClient::default();
26//! # let token = twitch_oauth2::AccessToken::new("validtoken".to_string());

41/// Query Parameters for [Get Chatters](super::get_user_chat_color)
42///
43/// [`get-user-chat-color`](https://dev.twitch.tv/docs/api/reference#get-user-chat-color)
44#[derive(PartialEq, Eq, Deserialize, Serialize, Clone, Debug)]
45#[cfg_attr(feature = "typed-builder", derive(typed_builder::TypedBuilder))]
46#[must_use]
47#[non_exhaustive]
48pub struct GetUserChatColorRequest<'a> {
49    /// The ID of the user whose color you want to get.
50    #[cfg_attr(
51        feature = "typed-builder",
52        builder(default_code = "types::Collection::default()", setter(into))
53    )]
54    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
55    // FIXME: This is essentially the same as borrow, but worse
56    #[cfg_attr(not(feature = "deser_borrow"), serde(bound(deserialize = "'de: 'a")))]
57    pub user_id: types::Collection<'a, types::UserId>,
58}
59
60impl<'a> GetUserChatColorRequest<'a> {
61    /// Get chat colors of specified users
62    ///
63    /// # Examples
64    ///
65    /// ```rust
66    /// use twitch_api::helix::chat::get_user_chat_color::GetUserChatColorRequest;
67    /// use twitch_api::types;
68    /// let ids: &[&types::UserIdRef] = &["1234".into()];
69    /// GetUserChatColorRequest::user_ids(ids);
70    /// ```
71    pub fn user_ids(user_ids: impl Into<types::Collection<'a, types::UserId>>) -> Self {
72        Self {
73            user_id: user_ids.into(),
74        }
75    }
76}

78/// Return Values for [Get Chatters](super::get_user_chat_color)
79///
80/// [`get-user-chat-color`](https://dev.twitch.tv/docs/api/reference#get-user-chat-color)
81#[derive(PartialEq, Eq, Deserialize, Serialize, Debug, Clone)]
82#[cfg_attr(feature = "deny_unknown_fields", serde(deny_unknown_fields))]
83#[non_exhaustive]
84pub struct UserChatColor {
85    /// The ID of the user.
86    pub user_id: types::UserId,
87    /// The user’s login name.
88    pub user_login: types::UserName,
89    /// The user’s display name.
90    pub user_name: types::DisplayName,
91    /// The Hex color code that the user uses in chat for their name.
92    ///
93    /// If the user hasn’t specified a color in their settings, the value is [`None`].
94    #[serde(
95        default,
96        deserialize_with = "crate::deserialize_none_from_empty_string"
97    )]
98    pub color: Option<types::HexColor>,
99}

113fn test_request() {
114    use helix::*;
115    let ids: &[&types::UserIdRef] = &["11111".into(), "44444".into()];
116    let req = GetUserChatColorRequest::user_ids(ids);
117
118    // From twitch docs
119    // FIXME: Example has ...
120    let data = br##"
121    {
122        "data": [
123          {
124            "user_id": "11111",
125            "user_name": "SpeedySpeedster1",
126            "user_login": "speedyspeedster1",
127            "color": "#9146FF"
128          },
129          {
130            "user_id": "44444",
131            "user_name": "SpeedySpeedster2",
132            "user_login": "speedyspeedster2",
133            "color": ""
134          }
135        ]
136      }
137"##
138    .to_vec();
139
140    let http_response = http::Response::builder().body(data).unwrap();
141
142    let uri = req.get_uri().unwrap();
143    assert_eq!(
144        uri.to_string(),
145        "https://api.twitch.tv/helix/chat/color?user_id=11111&user_id=44444"
146    );
147
148    let resp =
149        dbg!(GetUserChatColorRequest::parse_response(Some(req), &uri, http_response).unwrap());
150    assert!(resp.data.get(1).unwrap().color.is_none())
