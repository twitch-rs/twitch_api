32//!
33//! ```rust, no_run
34//! use twitch_api::helix::{self, chat::send_chat_message};

41//! let request = send_chat_message::SendChatMessageRequest::new();
42//! let body =
43//!     send_chat_message::SendChatMessageBody::new(

86pub struct SendChatMessageBody<'a> {
87    /// The ID of the broadcaster whose chat room the message will be sent to.
88    #[cfg_attr(feature = "typed-builder", builder(setter(into)))]
89    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
90    pub broadcaster_id: Cow<'a, types::UserIdRef>,
91    /// The ID of the user sending the message. This ID must match the user ID in the user access token.
92    #[cfg_attr(feature = "typed-builder", builder(setter(into)))]
93    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
94    pub sender_id: Cow<'a, types::UserIdRef>,
95    /// The message to send.
96    ///
97    /// The message is limited to a maximum of 500 characters.
98    /// Chat messages can also include emoticons.
99    /// To include emoticons, use the name of the emote. The names are case sensitive.
100    /// Don't include colons around the name (e.g., :bleedPurple:).
101    /// If Twitch recognizes the name, Twitch converts the name to the emote
102    /// before writing the chat message to the chat room.
103    #[cfg_attr(feature = "typed-builder", builder(setter(into)))]
104    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
105    pub message: Cow<'a, str>,
106    /// The ID of the chat message being replied to.
107    #[cfg_attr(feature = "typed-builder", builder(setter(into)))]
108    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
109    #[serde(skip_serializing_if = "Option::is_none")]
110    pub reply_parent_message_id: Option<Cow<'a, types::MsgIdRef>>,
111}

167/// Code for why a message was dropped.
168///
169/// See <https://dev.twitch.tv/docs/irc/msg-id/>
170#[derive(PartialEq, Eq, Deserialize, Serialize, Debug, Clone)]
171#[serde(rename_all = "snake_case")]
172#[non_exhaustive]
173pub enum ChatMessageDropCode {
174    /// You are permanently banned from talking in `<channel>`.
175    MsgBanned,
176    /// Your message was not sent because it contained too many unprocessable characters. If you believe this is an error, please rephrase and try again.
177    MsgBadCharacters,
178    /// Your message was not sent because your account is not in good standing in this channel.
179    MsgChannelBlocked,
180    /// This channel does not exist or has been suspended.
181    MsgChannelSuspended,
182    /// Your message was not sent because it is identical to the previous one you sent, less than 30 seconds ago.
183    MsgDuplicate,
184    /// This room is in emote-only mode. You can find your currently available emoticons using the smiley in the chat text area.
185    MsgEmoteonly,
186    /// This room is in `<duration>` followers-only mode. Follow `<channel>` to join the community! Note: These msg_followers tags are kickbacks to a user who does not meet the criteria; that is, does not follow or has not followed long enough.
187    MsgFollowersonly,
188    /// This room is in `<duration1>` followers-only mode. You have been following for `<duration2>`. Continue following to chat!
189    MsgFollowersonlyFollowed,
190    /// This room is in followers-only mode. Follow `<channel>` to join the community!
191    MsgFollowersonlyZero,
192    /// This room is in unique-chat mode and the message you attempted to send is not unique.
193    MsgR9k,
194    /// Your message was not sent because you are sending messages too quickly.
195    MsgRatelimit,
196    /// Hey! Your message is being checked by mods and has not been sent.
197    MsgRejected,
198    /// Your message wasnâ€™t posted due to conflicts with the channelâ€™s moderation settings.
199    MsgRejectedMandatory,
200    /// A verified phone number is required to chat in this channel. Please visit <https://www.twitch.tv/settings/security> to verify your phone number.
201    MsgRequiresVerifiedPhoneNumber,
202    /// This room is in slow mode and you are sending messages too quickly. You will be able to talk again in `<number>` seconds.
203    MsgSlowmode,
204    /// This room is in subscribers only mode. To talk, purchase a channel subscription at `https://www.twitch.tv/products/<broadcaster login name>/ticket?ref=subscriber_only_mode_chat`.
205    MsgSubsonly,
206    /// You donâ€™t have permission to perform that action.
207    MsgSuspended,
208    /// You are timed out for `<number>` more seconds.
209    MsgTimedout,
210    /// This room requires a verified account to chat. Please verify your account at <https://www.twitch.tv/settings/security>.
211    MsgVerifiedEmail,
212    /// An unknown drop-code.
213    #[serde(untagged)]
214    Unknown(String),
215}
