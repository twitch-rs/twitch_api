9//!
10//! ```rust

13//!     "1234", "5678",
14//! );
15//! ```

18//!
19//! Send the request to receive the response with [`HelixClient::req_post()`](helix::HelixClient::req_post).
20//!

41/// Query Parameters for [Get Pinned Chat Message](super::get_pinned_chat_message)
42///
43/// [`get-pinned-chat-message`](https://dev.twitch.tv/docs/api/reference#get-pinned-chat-message)
44#[derive(PartialEq, Eq, Deserialize, Serialize, Clone, Debug)]
45#[cfg_attr(feature = "typed-builder", derive(typed_builder::TypedBuilder))]
46#[must_use]
47#[non_exhaustive]
48pub struct GetPinnedChatMessageRequest<'a> {
49    /// The ID of the broadcaster that owns the chat room.
50    #[cfg_attr(feature = "typed-builder", builder(setter(into)))]
51    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
52    pub broadcaster_id: Cow<'a, types::UserIdRef>,
53    /// The ID of the broadcaster or a user that has permission to moderate the broadcaster’s chat room.
54    #[cfg_attr(feature = "typed-builder", builder(setter(into)))]
55    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
56    pub moderator_id: Cow<'a, types::UserIdRef>,
57}
58
59impl<'a> GetPinnedChatMessageRequest<'a> {
60    /// Get the currently pinned message for the specified broadcaster’s chat room.
61    pub fn new(
62        broadcaster_id: impl types::IntoCow<'a, types::UserIdRef> + 'a,
63        moderator_id: impl types::IntoCow<'a, types::UserIdRef> + 'a,
64    ) -> Self {
65        Self {
66            broadcaster_id: broadcaster_id.into_cow(),
67            moderator_id: moderator_id.into_cow(),
68        }
69    }
70}

72/// Return Values for [Get Pinned Chat Message](super::get_pinned_chat_message)
73///
74/// [`get-pinned-chat-message`](https://dev.twitch.tv/docs/api/reference#get-pinned-chat-message)
75#[derive(PartialEq, Eq, Deserialize, Serialize, Debug, Clone)]
76#[cfg_attr(feature = "deny_unknown_fields", serde(deny_unknown_fields))]
77#[non_exhaustive]
78pub struct PinnedChatMessage {
79    /// The ID of the pinned chat message.
80    pub message_id: types::MsgId,
81    /// The ID of the broadcaster.
82    pub broadcaster_id: types::UserId,
83    /// The ID of the user who sent the pinned message.
84    pub sender_user_id: types::UserId,
85    /// The login of the user who sent the pinned message.
86    pub sender_user_login: types::UserName,
87    /// The display name of the user who sent the pinned message.
88    pub sender_user_name: types::DisplayName,
89    /// The ID of the user who pinned the message.
90    pub pinned_by_user_id: types::UserId,
91    /// The login of the user who pinned the message.
92    pub pinned_by_user_login: types::UserName,
93    /// The display name of the user who pinned the message.
94    pub pinned_by_user_name: types::DisplayName,
95    /// The pinned message content.
96    pub message: crate::common::chat::Message,
97    /// RFC3339 timestamp of when the message was pinned.
98    pub starts_at: types::Timestamp,
99    /// RFC3339 expiry timestamp. [None] if pinned until stream ends.
100    pub ends_at: Option<types::Timestamp>,
101    /// RFC3339 timestamp of last update.
102    pub updated_at: types::Timestamp,
103}

117impl<'a> RequestGet for GetPinnedChatMessageRequest<'a> {
118    fn parse_inner_response(
119        request: Option<Self>,
120        uri: &http::Uri,
121        response: &str,
122        status: http::StatusCode,
123    ) -> Result<helix::Response<Self, <Self as Request>::Response>, helix::HelixRequestGetError>
124    where
125        Self: Sized,
126    {
127        let resp = match status {
128            http::StatusCode::OK => {
129                let resp: helix::InnerResponse<helix::request::ZeroOrOne<PinnedChatMessage>> =
130                    helix::parse_json(response, true).map_err(|e| {
131                        helix::HelixRequestGetError::DeserializeError(
132                            response.to_string(),
133                            e,
134                            uri.clone(),
135                            status,
136                        )
137                    })?;
138                resp.data.0
139            }
140            _ => {
141                return Err(helix::HelixRequestGetError::InvalidResponse {
142                    reason: "unexpected status code",
143                    response: response.to_string(),
144                    status,
145                    uri: uri.clone(),
146                })
147            }
148        };
149        Ok(helix::Response::with_data(resp, request))
150    }
151}
