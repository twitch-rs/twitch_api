4//! # Accessing the endpoint
5//!
6//! ## Request: [GetSharedChatSessionRequest]
7//!
8//! To use this endpoint, construct a [`GetSharedChatSessionRequest`] with the [`GetSharedChatSessionRequest::broadcaster_id`] method.
9//!
10//! ```rust
11//! use twitch_api::helix::chat::get_shared_chat_session;

13//!     get_shared_chat_session::GetSharedChatSessionRequest::broadcaster_id(
14//!         "12345",
15//!     );
16//! ```
17//!
18//! ## Response: [SharedChatSession]
19//!
20//! Send the request to receive the response with [`HelixClient::req_get()`](helix::HelixClient::req_get).
21//!
22//! ```rust, no_run
23//! use twitch_api::helix::{self, chat::get_shared_chat_session};

25//! # #[tokio::main]
26//! # async fn main() -> Result<(), Box<dyn std::error::Error + Send + Sync + 'static>> {
27//! # let client: helix::HelixClient<'static, client::DummyHttpClient> = helix::HelixClient::default();

130            }
131        };
132        Ok(helix::Response::with_data(resp, request))

136#[cfg(test)]
137#[test]
138fn test_request() {
139    use helix::*;
140    let req = GetSharedChatSessionRequest::broadcaster_id("198704263");
141
142    // From twitch docs (modified `updated_at`)
143    let data = br#"
144        {
145          "data": [
146            {
147              "session_id": "359bce59-fa4e-41a5-bd6f-9bc0c8360485",
148              "host_broadcaster_id": "198704263",
149              "participants": [{
150                  "broadcaster_id": "198704263"
151              }, {
152                  "broadcaster_id": "487263401"
153              }],
154              "created_at": "2024-09-29T19:45:37Z",
155              "updated_at": "2024-09-29T19:50:01Z"
156            }
157          ]
158        }
159        "#
160    .to_vec();
161
162    let http_response = http::Response::builder().body(data).unwrap();
163
164    let uri = req.get_uri().unwrap();
165    assert_eq!(
166        uri.to_string(),
167        "https://api.twitch.tv/helix/shared_chat/session?broadcaster_id=198704263"
168    );
169
170    let res = GetSharedChatSessionRequest::parse_response(Some(req), &uri, http_response).unwrap();
171    let res = res.data.unwrap();
172    assert_eq!(
173        res.session_id.as_str(),
174        "359bce59-fa4e-41a5-bd6f-9bc0c8360485"
175    );
176    assert_eq!(res.host_broadcaster_id.as_str(), "198704263");
177    assert_eq!(res.participants.len(), 2);
178    assert_eq!(res.participants[0].broadcaster_id.as_str(), "198704263");
179    assert_eq!(res.participants[1].broadcaster_id.as_str(), "487263401");
180    assert_eq!(res.created_at.as_str(), "2024-09-29T19:45:37Z");
181    assert_eq!(res.updated_at.as_str(), "2024-09-29T19:50:01Z");
182}
