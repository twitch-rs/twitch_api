1//! Gets all custom emotes for a specific Twitch channel including subscriber emotes, Bits tier emotes, and follower emotes.
2//! [`get-channel-emotes`](https://dev.twitch.tv/docs/api/reference#get-channel-emotes)
3//!
4//! # Accessing the endpoint
5//!
6//! ## Request: [GetChannelEmotesRequest]
7//!
8//! To use this endpoint, construct a [`GetChannelEmotesRequest`] with the [`GetChannelEmotesRequest::broadcaster_id()`] method.
9//!
10//! ```rust
11//! use twitch_api::helix::chat::get_channel_emotes;
12//! let request =
13//!     get_channel_emotes::GetChannelEmotesRequest::broadcaster_id("1234");
14//! ```
15//!
16//! ## Response: [ChannelEmote]
17//!
18//! Send the request to receive the response with [`HelixClient::req_get()`](helix::HelixClient::req_get).
19//!
20//! ```rust, no_run
21//! use twitch_api::helix::{self, chat::get_channel_emotes};
22//! # use twitch_api::client;
23//! # #[tokio::main]
24//! # async fn main() -> Result<(), Box<dyn std::error::Error + Send + Sync + 'static>> {
25//! # let client: helix::HelixClient<'static, client::DummyHttpClient> = helix::HelixClient::default();
26//! # let token = twitch_oauth2::AccessToken::new("validtoken".to_string());
27//! # let token = twitch_oauth2::UserToken::from_existing(&client, token, None, None).await?;
28//! let request = get_channel_emotes::GetChannelEmotesRequest::broadcaster_id("1234");
29//! let response: Vec<helix::chat::ChannelEmote> = client.req_get(request, &token).await?.data;
30//! # Ok(())
31//! # }
32//! ```
33//!
34//! You can also get the [`http::Request`] with [`request.create_request(&token, &client_id)`](helix::RequestGet::create_request)
35//! and parse the [`http::Response`] with [`GetChannelEmotesRequest::parse_response(None, &request.get_uri(), response)`](GetChannelEmotesRequest::parse_response)

40/// Query Parameters for [Get Channel Emotes](super::get_channel_emotes)
41///
42/// [`get-channel-emotes`](https://dev.twitch.tv/docs/api/reference#get-channel-emotes)
43#[derive(PartialEq, Eq, Deserialize, Serialize, Clone, Debug)]
44#[cfg_attr(feature = "typed-builder", derive(typed_builder::TypedBuilder))]
45#[must_use]
46#[non_exhaustive]
47pub struct GetChannelEmotesRequest<'a> {
48    /// The broadcaster whose emotes are being requested.
49    #[cfg_attr(feature = "typed-builder", builder(setter(into)))]
50    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
51    pub broadcaster_id: Cow<'a, types::UserIdRef>,
52}
53
54impl<'a> GetChannelEmotesRequest<'a> {
55    /// Get emotes in a specific broadcasters channel.
56    pub fn broadcaster_id(broadcaster_id: impl types::IntoCow<'a, types::UserIdRef> + 'a) -> Self {
57        Self {
58            broadcaster_id: broadcaster_id.into_cow(),
59        }
60    }
61}

63/// Return Values for [Get Channel Emotes](super::get_channel_emotes)

66pub type GetChannelEmotesResponse = ChannelEmote;
