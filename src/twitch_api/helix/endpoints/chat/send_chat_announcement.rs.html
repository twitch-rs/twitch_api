1//! Sends an announcement to the broadcaster’s chat room.
2//! [`send-chat-announcement`](https://dev.twitch.tv/docs/api/reference#send-chat-announcement)
3//!
4//! # Accessing the endpoint
5//!
6//! ## Request: [SendChatAnnouncementRequest]
7//!
8//! To use this endpoint, construct a [`SendChatAnnouncementRequest`] with the [`SendChatAnnouncementRequest::new()`] method.
9//!
10//! ```rust
11//! use twitch_api::helix::chat::send_chat_announcement;
12//! let request = send_chat_announcement::SendChatAnnouncementRequest::new(
13//!     "1234", "5678",
14//! );
15//! ```
16//!
17//! ## Body: [SendChatAnnouncementBody]
18//!
19//! We also need to provide a body to the request containing what we want to change.
20//!
21//! ```
22//! # use twitch_api::helix::chat::send_chat_announcement;
23//! let body = send_chat_announcement::SendChatAnnouncementBody::new(
24//!     "Hello chat!",
25//!     "purple",
26//! )
27//! .unwrap();
28//! ```
29//!
30//! ## Response: [SendChatAnnouncementResponse]
31//!
32//!
33//! Send the request to receive the response with [`HelixClient::req_post()`](helix::HelixClient::req_post).
34//!
35//! ```rust, no_run
36//! use twitch_api::helix::{self, chat::send_chat_announcement};
37//! # use twitch_api::client;
38//! # #[tokio::main]
39//! # async fn main() -> Result<(), Box<dyn std::error::Error + Send + Sync + 'static>> {
40//! # let client: helix::HelixClient<'static, client::DummyHttpClient> = helix::HelixClient::default();
41//! # let token = twitch_oauth2::AccessToken::new("validtoken".to_string());
42//! # let token = twitch_oauth2::UserToken::from_existing(&client, token, None, None).await?;
43//! let request = send_chat_announcement::SendChatAnnouncementRequest::new("1234", "5678");
44//! let body = send_chat_announcement::SendChatAnnouncementBody::new(
45//!     "Hello chat!",
46//!     "purple",
47//! ).unwrap();
48//! let response: helix::chat::SendChatAnnouncementResponse = client.req_post(request, body, &token).await?.data;
49//! # Ok(())
50//! # }
51//! ```
52//!
53//! You can also get the [`http::Request`] with [`request.create_request(&token, &client_id)`](helix::RequestPost::create_request)
54//! and parse the [`http::Response`] with [`SendChatAnnouncementRequest::parse_response(None, &request.get_uri(), response)`](SendChatAnnouncementRequest::parse_response)

58/// Query Parameters for [Send Chat Announcement](super::send_chat_announcement)
59///

62#[cfg_attr(feature = "typed-builder", derive(typed_builder::TypedBuilder))]
63#[must_use]
64#[non_exhaustive]
65pub struct SendChatAnnouncementRequest<'a> {
66    /// The ID of the broadcaster that owns the chat room to send the announcement to.
67    #[cfg_attr(feature = "typed-builder", builder(setter(into)))]
68    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]

70    /// The ID of a user who has permission to moderate the broadcaster’s chat room.

73    #[cfg_attr(feature = "typed-builder", builder(setter(into)))]
74    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
75    pub moderator_id: Cow<'a, types::UserIdRef>,
76}

91/// Body Parameters for [Send Chat Announcement](super::send_chat_announcement)

96#[non_exhaustive]
97pub struct SendChatAnnouncementBody<'a> {
98    /// The announcement to make in the broadcaster’s chat room. Announcements are limited to a maximum of 500 characters; announcements longer than 500 characters are truncated.
99    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
100    pub message: Cow<'a, str>,
101    // FIXME: Enumify?
102    /// The color used to highlight the announcement. Possible case-sensitive values are:
103    ///
104    /// * blue
105    /// * green
106    /// * orange
107    /// * purple
108    /// * primary (default)
109    ///
110    /// If color is set to primary or is not set, the channel’s accent color is used to highlight the announcement (see Profile Accent Color under [profile settings](https://www.twitch.tv/settings/profile), Channel and Videos, and Brand).
111    #[cfg_attr(feature = "typed-builder", builder(setter(into)))]
112    pub color: AnnouncementColor,
113}
