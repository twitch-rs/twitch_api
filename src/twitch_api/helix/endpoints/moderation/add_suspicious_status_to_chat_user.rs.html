150fn test_request() {
151    use helix::*;
152    let req = AddSuspiciousStatusToChatUserRequest::new("141981764", "12826");
153
154    let body =
155        AddSuspiciousStatusToChatUserBody::new("9876", AddedSuspiciousUserStatus::Restricted);
156
157    assert_eq!(
158        std::str::from_utf8(&body.try_to_body().unwrap()).unwrap(),
159        r#"{"user_id":"9876","status":"RESTRICTED"}"#
160    );
161
162    dbg!(req.create_request(body, "token", "clientid").unwrap());
163
164    // From twitch docs
165    let data = br#"
166    {
167        "data": [
168            {
169                "user_id": "9876",
170                "broadcaster_id": "141981764",
171                "moderator_id": "12826",
172                "updated_at": "2025-12-01T23:08:18+00:00",
173                "status": "RESTRICTED",
174                "types": [
175                    "MANUALLY_ADDED"
176                ]
177            }
178        ]
179    }
180    "#
181    .to_vec();
182
183    let http_response = http::Response::builder().body(data).unwrap();
184
185    let uri = req.get_uri().unwrap();
186    assert_eq!(
187        uri.to_string(),
188        "https://api.twitch.tv/helix/moderation/suspicious_users?broadcaster_id=141981764&moderator_id=12826"
189    );
190
191    dbg!(
192        AddSuspiciousStatusToChatUserRequest::parse_response(Some(req), &uri, http_response)
193            .unwrap()
194    );
