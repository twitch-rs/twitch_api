1//! Resolves an unban request by approving or denying it.
2//!
3//! [`resolve-unban-requests`](https://dev.twitch.tv/docs/api/reference#resolve-unban-requests)
4//!
5//! # Accessing the endpoint
6//!
7//! ## Request: [ResolveUnbanRequest]
8//!
9//! To use this endpoint, construct an [`ResolveUnbanRequest`] with the [`ResolveUnbanRequest::new()`] method.
10//!
11//! ```rust
12//! use twitch_api::helix::moderation::resolve_unban_request;
13//! let request = resolve_unban_request::ResolveUnbanRequest::approve(
14//!     "123",
15//!     "456",
16//!     "123-456-789",
17//! )
18//! .resolution_text("something");
19//! ```
20//!
21//! ## Response: [UnbanRequest]
22//!
23//! Send the request to receive the response with [`HelixClient::req_patch()`](helix::HelixClient::req_patch).
24//!
25//! ```rust, no_run
26//! use twitch_api::helix::{self, moderation::resolve_unban_request};
27//! # use twitch_api::client;
28//! # #[tokio::main]
29//! # async fn main() -> Result<(), Box<dyn std::error::Error + Send + Sync + 'static>> {
30//! # let client: helix::HelixClient<'static, client::DummyHttpClient> = helix::HelixClient::default();
31//! # let token = twitch_oauth2::AccessToken::new("validtoken".to_string());
32//! # let token = twitch_oauth2::UserToken::from_existing(&client, token, None, None).await?;
33//! let request = resolve_unban_request::ResolveUnbanRequest::approve(
34//!     "123",
35//!     "456",
36//!     "123-456-789",
37//! );
38//! let response: helix::moderation::UnbanRequest = client.req_patch(request, helix::EmptyBody, &token).await?.data;
39//! # Ok(())
40//! # }
41//! ```
42//!
43//! You can also get the [`http::Request`] with [`request.create_request(&token, &client_id)`](helix::RequestPut::create_request)
44//! and parse the [`http::Response`] with [`ResolveUnbanRequest::parse_response(None, &request.get_uri(), response)`](ResolveUnbanRequest::parse_response)

51/// Query Parameters for [Resolve Unban Request](super::resolve_unban_request)
52///
53/// [`resolve-unban-requests`](https://dev.twitch.tv/docs/api/reference#resolve-unban-requests)
54#[derive(PartialEq, Eq, Deserialize, Serialize, Clone, Debug)]
55#[cfg_attr(feature = "typed-builder", derive(typed_builder::TypedBuilder))]
56#[non_exhaustive]
57pub struct ResolveUnbanRequest<'a> {
58    /// The ID of the broadcaster whose channel is approving or denying the unban request.
59    #[cfg_attr(feature = "typed-builder", builder(setter(into)))]
60    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
61    #[cfg_attr(not(feature = "deser_borrow"), serde(bound(deserialize = "'de: 'a")))]
62    pub broadcaster_id: Cow<'a, types::UserIdRef>,
63    /// The ID of the broadcaster or a user that has permission to moderate the broadcasterâ€™s unban requests. This ID must match the user ID in the user access token.
64    #[cfg_attr(feature = "typed-builder", builder(setter(into)))]
65    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
66    #[cfg_attr(not(feature = "deser_borrow"), serde(bound(deserialize = "'de: 'a")))]
67    pub moderator_id: Cow<'a, types::UserIdRef>,
68    /// The ID of the Unban Request to resolve.
69    #[cfg_attr(feature = "typed-builder", builder(setter(into)))]
70    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
71    #[cfg_attr(not(feature = "deser_borrow"), serde(bound(deserialize = "'de: 'a")))]
72    pub unban_request_id: Cow<'a, types::UnbanRequestIdRef>,
73    /// Resolution status.
74    ///
75    /// Only [Approved](UnbanRequestStatus::Approved) and [Denied](UnbanRequestStatus::Denied) are accepted.
76    pub status: UnbanRequestStatus,
77    #[cfg_attr(feature = "typed-builder", builder(setter(into)))]
78    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
79    #[cfg_attr(not(feature = "deser_borrow"), serde(bound(deserialize = "'de: 'a")))]
80    /// Message supplied by the unban request resolver. The message is limited to a maximum of 500 characters.
81    pub resolution_text: Option<Cow<'a, str>>,
82}

84impl<'a> ResolveUnbanRequest<'a> {
85    /// Approve an unban request
86    pub fn approve(
87        broadcaster_id: impl types::IntoCow<'a, types::UserIdRef> + 'a,
88        moderator_id: impl types::IntoCow<'a, types::UserIdRef> + 'a,
89        unban_request_id: impl types::IntoCow<'a, types::UnbanRequestIdRef> + 'a,
90    ) -> Self {
91        Self {
92            broadcaster_id: broadcaster_id.into_cow(),
93            moderator_id: moderator_id.into_cow(),
94            unban_request_id: unban_request_id.into_cow(),
95            status: UnbanRequestStatus::Approved,
96            resolution_text: None,
97        }
98    }
99
100    /// Deny an unban request
101    pub fn deny(
102        broadcaster_id: impl types::IntoCow<'a, types::UserIdRef> + 'a,
103        moderator_id: impl types::IntoCow<'a, types::UserIdRef> + 'a,
104        unban_request_id: impl types::IntoCow<'a, types::UnbanRequestIdRef> + 'a,
105    ) -> Self {
106        Self {
107            broadcaster_id: broadcaster_id.into_cow(),
108            moderator_id: moderator_id.into_cow(),
109            unban_request_id: unban_request_id.into_cow(),
110            status: UnbanRequestStatus::Denied,
111            resolution_text: None,
112        }
113    }
114
115    /// Resolve an unban request
116    ///
117    /// Only [Approved](UnbanRequestStatus::Approved) and [Denied](UnbanRequestStatus::Denied) are accepted as the status.
118    pub fn new(
119        broadcaster_id: impl types::IntoCow<'a, types::UserIdRef> + 'a,
120        moderator_id: impl types::IntoCow<'a, types::UserIdRef> + 'a,
121        unban_request_id: impl types::IntoCow<'a, types::UnbanRequestIdRef> + 'a,
122        status: UnbanRequestStatus,
123    ) -> Self {
124        Self {
125            broadcaster_id: broadcaster_id.into_cow(),
126            moderator_id: moderator_id.into_cow(),
127            unban_request_id: unban_request_id.into_cow(),
128            status,
129            resolution_text: None,
130        }
131    }
132
133    /// Set the resolution text for a request
134    pub fn resolution_text(mut self, text: impl types::IntoCow<'a, str> + 'a) -> Self {
135        self.resolution_text = Some(text.into_cow());
136        self
137    }
138}

165#[cfg(test)]
166#[test]
167fn test_request() {
168    use helix::*;
169    // modified from twitch example (they specified no resolution and the wrong moderator)
170    let req = ResolveUnbanRequest::approve(
171        "274637212",
172        "141981764",
173        "92af127c-7326-4483-a52b-b0da0be61c01",
174    )
175    .resolution_text("okay");
176
177    req.create_request(helix::EmptyBody, "token", "clientid")
178        .unwrap();
179
180    // Own response because the docs are wrong
181    let data = br#"
182    {
183        "data": [
184            {
185                "broadcaster_id": "129546453",
186                "broadcaster_login": "nerixyz",
187                "broadcaster_name": "nerixyz",
188                "created_at": "2024-10-12T18:34:38Z",
189                "id": "1dfff107-17fc-44cb-9f64-570a33757ac0",
190                "moderator_id": "129546453",
191                "moderator_login": "nerixyz",
192                "moderator_name": "nerixyz",
193                "resolution_text": "",
194                "resolved_at": "2024-10-12T18:45:47Z",
195                "status": "approved",
196                "text": "My unban request text",
197                "user_id": "489584266",
198                "user_login": "uint128",
199                "user_name": "uint128"
200            }
201        ]
202    }
203    "#
204    .to_vec();
205
206    let http_response = http::Response::builder().status(200).body(data).unwrap();
207
208    let uri = req.get_uri().unwrap();
209    assert_eq!(
210        uri.to_string(),
211        "https://api.twitch.tv/helix/moderation/unban_requests?broadcaster_id=274637212&moderator_id=141981764&unban_request_id=92af127c-7326-4483-a52b-b0da0be61c01&status=approved&resolution_text=okay"
212    );
213
214    let res = ResolveUnbanRequest::parse_response(Some(req), &uri, http_response)
215        .unwrap()
216        .data;
217    assert_eq!(res.status, UnbanRequestStatus::Approved);
218    assert_eq!(res.resolution_text.as_deref(), Some(""));
219}
