1//! Removes a single chat message or all chat messages from the broadcaster’s chat room.
2//! [`delete-chat-messages`](https://dev.twitch.tv/docs/api/reference#delete-chat-messages)
3//!
4//! # Accessing the endpoint
5//!
6//! ## Request: [DeleteChatMessagesRequest]
7//!
8//! To use this endpoint, construct a [`DeleteChatMessagesRequest`] with the [`DeleteChatMessagesRequest::new()`] method.
9//!
10//! ```rust
11//! use twitch_api::helix::moderation::delete_chat_messages;
12//! let request =
13//!     delete_chat_messages::DeleteChatMessagesRequest::new("1234", "5678")
14//!         .message_id("abc-123-def");
15//! ```
16//!
17//! ## Response: [DeleteChatMessagesResponse]
18//!
19//! Send the request to receive the response with [`HelixClient::req_delete()`](helix::HelixClient::req_delete).
20//!
21//! ```rust, no_run
22//! use twitch_api::helix::{self, moderation::delete_chat_messages};
23//! # use twitch_api::client;
24//! # #[tokio::main]
25//! # async fn main() -> Result<(), Box<dyn std::error::Error + Send + Sync + 'static>> {
26//! # let client: helix::HelixClient<'static, client::DummyHttpClient> = helix::HelixClient::default();

28//! # let token = twitch_oauth2::UserToken::from_existing(&client, token, None, None).await?;

41/// Query Parameters for [Delete Chat Messages](super::delete_chat_messages)
42///
43/// [`delete-chat-messages`](https://dev.twitch.tv/docs/api/reference#delete-chat-messages)
44#[derive(PartialEq, Eq, Deserialize, Serialize, Clone, Debug)]
45#[cfg_attr(feature = "typed-builder", derive(typed_builder::TypedBuilder))]
46#[must_use]
47#[non_exhaustive]
48pub struct DeleteChatMessagesRequest<'a> {
49    /// The ID of the broadcaster that owns the chat room to remove messages from.
50    #[cfg_attr(feature = "typed-builder", builder(setter(into)))]
51    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
52    pub broadcaster_id: Cow<'a, types::UserIdRef>,
53    /// The ID of a user that has permission to moderate the broadcaster’s chat room.
54    ///
55    /// This ID must match the user ID in the OAuth token. If the broadcaster wants to remove messages themselves, set this parameter to the broadcaster’s ID, too.
56    #[cfg_attr(feature = "typed-builder", builder(setter(into)))]
57    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
58    pub moderator_id: Cow<'a, types::UserIdRef>,
59    /// The ID of the message to remove.
60    ///
61    /// The id tag in the PRIVMSG contains the message’s ID (see [PRIVMSG Tags](https://dev.twitch.tv/docs/irc/tags#privmsg-tags)).
62    ///
63    /// # Restrictions
64    ///
65    /// The message must have been created within the last 6 hours.
66    /// The message must not belong to the broadcaster.
67    /// The message must not belong to another moderator.
68    ///
69    /// If not specified, the request removes all messages in the broadcaster’s chat room.
70    #[cfg_attr(feature = "typed-builder", builder(default, setter(into)))]
71    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
72    pub message_id: Option<Cow<'a, types::MsgIdRef>>,
73}
74
75impl<'a> DeleteChatMessagesRequest<'a> {
76    /// Remove chat message(s)
77    pub fn new(
78        broadcaster_id: impl types::IntoCow<'a, types::UserIdRef> + 'a,
79        moderator_id: impl types::IntoCow<'a, types::UserIdRef> + 'a,
80    ) -> Self {
81        Self {
82            broadcaster_id: broadcaster_id.into_cow(),
83            moderator_id: moderator_id.into_cow(),
84            message_id: None,
85        }
86    }
87
88    /// A specific message to remove
89    pub fn message_id(mut self, message_id: impl types::IntoCow<'a, types::MsgIdRef> + 'a) -> Self {
90        self.message_id = Some(message_id.into_cow());
91        self
92    }
93}
