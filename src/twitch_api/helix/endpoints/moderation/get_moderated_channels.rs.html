1//! Gets a list of channels that the specified user has moderator privileges in.
2//! [`get-moderated-channels`](https://dev.twitch.tv/docs/api/reference#get-moderated-channels)
3//!
4//! # Accessing the endpoint
5//!
6//! ## Request: [GetModeratedChannelsRequest]
7//!
8//! To use this endpoint, construct a [`GetModeratedChannelsRequest`] with the [`GetModeratedChannelsRequest::user_id()`] method.
9//!
10//! ```rust
11//! use twitch_api::helix::moderation::get_moderated_channels;
12//! let request =
13//!     get_moderated_channels::GetModeratedChannelsRequest::user_id("1234");
14//! ```
15//!
16//! ## Response: [ModeratedChannel]
17//!
18//! Send the request to receive the response with [`HelixClient::req_get()`](helix::HelixClient::req_get).
19//!
20//! ```rust, no_run
21//! use twitch_api::helix::{self, moderation::get_moderated_channels};
22//! # use twitch_api::client;
23//! # #[tokio::main]
24//! # async fn main() -> Result<(), Box<dyn std::error::Error + Send + Sync + 'static>> {
25//! # let client: helix::HelixClient<'static, client::DummyHttpClient> = helix::HelixClient::default();
26//! # let token = twitch_oauth2::AccessToken::new("validtoken".to_string());
27//! # let token = twitch_oauth2::UserToken::from_existing(&client, token, None, None).await?;
28//! let request = get_moderated_channels::GetModeratedChannelsRequest::user_id("1234");
29//! let response: Vec<get_moderated_channels::ModeratedChannel> = client.req_get(request, &token).await?.data;
30//! # Ok(())
31//! # }
32//! ```

34//! You can also get the [`http::Request`] with [`request.create_request(&token, &client_id)`](helix::RequestGet::create_request)
35//! and parse the [`http::Response`] with [`GetModeratedChannelsRequest::parse_response(None, &request.get_uri(), response)`](GetModeratedChannelsRequest::parse_response)

39/// Query Parameters for [Get Moderated Channels](super::get_moderated_channels)
40///
41/// [`get-moderated-channels`](https://dev.twitch.tv/docs/api/reference#get-moderated-channels)
42#[derive(PartialEq, Eq, Deserialize, Serialize, Clone, Debug)]
43#[cfg_attr(feature = "typed-builder", derive(typed_builder::TypedBuilder))]
44#[must_use]
45#[non_exhaustive]
46pub struct GetModeratedChannelsRequest<'a> {
47    /// A user’s ID. Returns the list of channels that this user has moderator privileges in. This ID must match the user ID in the user OAuth token.
48    #[cfg_attr(feature = "typed-builder", builder(setter(into)))]
49    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
50    pub user_id: Cow<'a, types::UserIdRef>,
51    /// The cursor used to get the next page of results. The Pagination object in the response contains the cursor’s value.
52    #[cfg_attr(feature = "typed-builder", builder(default))]
53    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
54    pub after: Option<Cow<'a, helix::CursorRef>>,
55    /// The maximum number of items to return per page in the response. Limit: 100. Default: 20.
56    #[cfg_attr(feature = "typed-builder", builder(setter(into), default))]
57    pub first: Option<usize>,
58}
59
60impl<'a> GetModeratedChannelsRequest<'a> {
61    /// Get Moderated Channels for an authenticated user.
62    pub fn user_id(user_id: impl types::IntoCow<'a, types::UserIdRef> + 'a) -> Self {
63        Self {
64            user_id: user_id.into_cow(),
65            after: Default::default(),
66            first: Default::default(),
67        }
68    }
69
70    /// Set amount of results returned per page.
71    pub fn first(mut self, first: usize) -> Self {
72        self.first = Some(first);
73        self
74    }
75}

77/// Return Values for [Get Moderated Channels](super::get_moderated_channels)
78///
79/// [`get-moderated-channels`](https://dev.twitch.tv/docs/api/reference#get-moderated-channels)
80#[derive(PartialEq, Eq, Deserialize, Serialize, Debug, Clone)]
81#[cfg_attr(feature = "deny_unknown_fields", serde(deny_unknown_fields))]
82#[non_exhaustive]
83pub struct ModeratedChannel {
84    /// An ID that uniquely identifies the channel this user can moderate.
85    pub broadcaster_id: types::UserId,
86    /// The channel’s login name.
87    pub broadcaster_login: types::UserName,
88    /// The channels’ display name.
89    pub broadcaster_name: types::DisplayName,
90}

103impl helix::Paginated for GetModeratedChannelsRequest<'_> {

109#[cfg(test)]
110#[test]
111fn test_request() {
112    use helix::*;
113    let req = GetModeratedChannelsRequest::user_id("931931");
114
115    // From twitch docs
116    let data = br#"
117    {
118        "data": [
119            {
120                "broadcaster_id" : "12345",
121                "broadcaster_login" : "grateful_broadcaster",
122                "broadcaster_name" : "Grateful_Broadcaster"
123            },
124            {
125                "broadcaster_id" : "98765",
126                "broadcaster_login" : "bashfulgamer",
127                "broadcaster_name" : "BashfulGamer"
128            }
129        ],
130        "pagination" : {
131            "cursor" : "eyJiIjpudWxsLCJhIjp7IkN1cnNvciI6IjEwMDQ3MzA2NDo4NjQwNjU3MToxSVZCVDFKMnY5M1BTOXh3d1E0dUdXMkJOMFcifX0"
132        }
133    }
134"#
135        .to_vec();
136
137    let http_response = http::Response::builder().body(data).unwrap();
138
139    let uri = req.get_uri().unwrap();
140    assert_eq!(
141        uri.to_string(),
142        "https://api.twitch.tv/helix/moderation/channels?user_id=931931"
143    );
144
145    let res = GetModeratedChannelsRequest::parse_response(Some(req), &uri, http_response)
146        .unwrap()
147        .data;
148    assert_eq!(res.len(), 2);
149
150    let (first, second) = (&res[0], &res[1]);
151    assert_eq!(first.broadcaster_id.as_str(), "12345");
152    assert_eq!(first.broadcaster_login.as_str(), "grateful_broadcaster");
153    assert_eq!(first.broadcaster_name.as_str(), "Grateful_Broadcaster");
154    assert_eq!(second.broadcaster_id.as_str(), "98765");
155    assert_eq!(second.broadcaster_login.as_str(), "bashfulgamer");
156    assert_eq!(second.broadcaster_name.as_str(), "BashfulGamer");
157}
