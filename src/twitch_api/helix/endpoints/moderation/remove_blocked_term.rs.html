1//! Removes the word or phrase that the broadcaster is blocking users from using in their chat room.
2//! [`remove-blocked-term`](https://dev.twitch.tv/docs/api/reference#remove-blocked-term)
3//!
4//! # Accessing the endpoint
5//!
6//! ## Request: [RemoveBlockedTermRequest]
7//!
8//! To use this endpoint, construct a [`RemoveBlockedTermRequest`] with the [`RemoveBlockedTermRequest::new()`] method.
9//!
10//! ```rust
11//! use twitch_api::helix::moderation::remove_blocked_term;
12//! let request = remove_blocked_term::RemoveBlockedTermRequest::new(
13//!     "1234", "5678", "DEADBEEF",
14//! );
15//! ```
16//!
17//! ## Response: [RemoveBlockedTerm]
18//!
19//! Send the request to receive the response with [`HelixClient::req_delete()`](helix::HelixClient::req_delete).
20//!
21//! ```rust, no_run
22//! use twitch_api::helix::{self, moderation::remove_blocked_term};
23//! # use twitch_api::client;
24//! # #[tokio::main]
25//! # async fn main() -> Result<(), Box<dyn std::error::Error + Send + Sync + 'static>> {
26//! # let client: helix::HelixClient<'static, client::DummyHttpClient> = helix::HelixClient::default();
27//! # let token = twitch_oauth2::AccessToken::new("validtoken".to_string());
28//! # let token = twitch_oauth2::UserToken::from_existing(&client, token, None, None).await?;
29//! let request = remove_blocked_term::RemoveBlockedTermRequest::new("1234", "5678", "DEADBEEF");
30//! let response: remove_blocked_term::RemoveBlockedTerm = client.req_delete(request, &token).await?.data;
31//! # Ok(())
32//! # }
33//! ```
34//!
35//! You can also get the [`http::Request`] with [`request.create_request(&token, &client_id)`](helix::RequestDelete::create_request)
36//! and parse the [`http::Response`] with [`RemoveBlockedTermRequest::parse_response(None, &request.get_uri(), response)`](RemoveBlockedTermRequest::parse_response)

40/// Query Parameters for [Remove Blocked Terms](super::remove_blocked_term)
41///
42/// [`remove-blocked-term`](https://dev.twitch.tv/docs/api/reference#remove-blocked-term)
43#[derive(PartialEq, Eq, Deserialize, Serialize, Clone, Debug)]
44#[cfg_attr(feature = "typed-builder", derive(typed_builder::TypedBuilder))]
45#[must_use]
46#[non_exhaustive]
47pub struct RemoveBlockedTermRequest<'a> {
48    /// The ID of the broadcaster that owns the list of blocked terms.
49    #[cfg_attr(feature = "typed-builder", builder(setter(into)))]
50    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
51    pub broadcaster_id: Cow<'a, types::UserIdRef>,
52    /// The ID of a user that has permission to moderate the broadcaster’s chat room. This ID must match the user ID associated with the user OAuth token.
53    /// If the broadcaster wants to delete the blocked term (instead of having the moderator do it), set this parameter to the broadcaster’s ID, too.
54    #[cfg_attr(feature = "typed-builder", builder(setter(into)))]
55    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
56    pub moderator_id: Cow<'a, types::UserIdRef>,
57    /// The ID of the blocked term you want to delete.
58    #[cfg_attr(feature = "typed-builder", builder(setter(into)))]
59    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
60    pub id: Cow<'a, types::BlockedTermIdRef>,
61}
62
63impl<'a> RemoveBlockedTermRequest<'a> {
64    /// Remove blocked term
65    pub fn new(
66        broadcaster_id: impl types::IntoCow<'a, types::UserIdRef> + 'a,
67        moderator_id: impl types::IntoCow<'a, types::UserIdRef> + 'a,
68        id: impl types::IntoCow<'a, types::BlockedTermIdRef> + 'a,
69    ) -> Self {
70        Self {
71            broadcaster_id: broadcaster_id.into_cow(),
72            moderator_id: moderator_id.into_cow(),
73            id: id.into_cow(),
74        }
75    }
76}

99impl RequestDelete for RemoveBlockedTermRequest<'_> {
100    fn parse_inner_response(
101        request: Option<Self>,
102        uri: &http::Uri,
103        response: &str,
104        status: http::StatusCode,
105    ) -> Result<helix::Response<Self, Self::Response>, helix::HelixRequestDeleteError>
106    where
107        Self: Sized,
108    {
109        match status {
110            http::StatusCode::NO_CONTENT => Ok(helix::Response::with_data(
111                RemoveBlockedTerm::Success,
112                request,
113            )),
114            _ => Err(helix::HelixRequestDeleteError::InvalidResponse {
115                reason: "unexpected status",
116                response: response.to_string(),
117                status,
118                uri: uri.clone(),
119            }),
120        }
121    }
122}
