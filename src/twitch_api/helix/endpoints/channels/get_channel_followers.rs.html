14//!         "1234",

43/// Query Parameters for [Get Channel Followers](super::get_channel_followers)
44///
45/// [`get-channel-followers`](https://dev.twitch.tv/docs/api/reference#get-channel-followers)
46#[derive(PartialEq, Eq, Deserialize, Serialize, Clone, Debug)]
47#[cfg_attr(feature = "typed-builder", derive(typed_builder::TypedBuilder))]
48#[must_use]
49#[non_exhaustive]
50pub struct GetChannelFollowersRequest<'a> {
51    /// A user’s ID. Use this parameter to see whether the user follows this broadcaster.
52    /// If specified, the response contains this user if they follow the broadcaster.
53    /// If not specified, the response contains all users that follow the broadcaster.
54    #[cfg_attr(feature = "typed-builder", builder(default, setter(into)))]
55    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
56    pub user_id: Option<Cow<'a, types::UserIdRef>>,
57    /// Broadcaster’s user ID associated with the channel.
58    #[cfg_attr(feature = "typed-builder", builder(setter(into)))]
59    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
60    pub broadcaster_id: Cow<'a, types::UserIdRef>,
61    /// Cursor for forward pagination: tells the server where to start fetching the next set of results, in a multi-page response. The cursor value specified here is from the pagination response field of a prior query.
62    #[cfg_attr(feature = "typed-builder", builder(default))]
63    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
64    pub after: Option<Cow<'a, helix::CursorRef>>,
65    /// Maximum number of objects to return. Maximum: 100. Default: 20.
66    #[cfg_attr(feature = "typed-builder", builder(default, setter(into)))]
67    pub first: Option<usize>,
68}

70impl<'a> GetChannelFollowersRequest<'a> {
71    /// Get specified broadcasters channel editors
72    ///
73    /// # Examples
74    ///
75    /// ```rust
76    /// use twitch_api::helix::channels::GetChannelFollowersRequest;
77    /// let request = GetChannelFollowersRequest::broadcaster_id("1234");
78    /// ```
79    pub fn broadcaster_id(broadcaster_id: impl types::IntoCow<'a, types::UserIdRef> + 'a) -> Self {
80        Self {
81            broadcaster_id: broadcaster_id.into_cow(),
82            user_id: None,
83            after: None,
84            first: None,
85        }
86    }
87
88    /// Set amount of results returned per page.
89    pub fn first(mut self, first: usize) -> Self {
90        self.first = Some(first);
91        self
92    }
93
94    /// Check if this user id is following the broadcaster
95    pub fn user_id(self, user_id: impl types::IntoCow<'a, types::UserIdRef> + 'a) -> Self {
96        Self {
97            user_id: Some(user_id.into_cow()),
98            ..self
99        }
100    }
101}

103/// Return Values for [Get Channel Followers](super::get_channel_followers)
104///
105/// [`get-channel-followers`](https://dev.twitch.tv/docs/api/reference#get-channel-followers)
106#[derive(PartialEq, Eq, Deserialize, Serialize, Debug, Clone)]
107#[cfg_attr(feature = "deny_unknown_fields", serde(deny_unknown_fields))]
108#[non_exhaustive]
109pub struct Follower {
110    /// The UTC timestamp when the user started following the broadcaster.
111    pub followed_at: types::Timestamp,
112    /// An ID that uniquely identifies the user that’s following the broadcaster.
113    pub user_id: types::UserId,
114    /// The user’s login name.
115    pub user_login: types::UserName,
116    /// The user’s display name.
117    pub user_name: types::DisplayName,
118}
