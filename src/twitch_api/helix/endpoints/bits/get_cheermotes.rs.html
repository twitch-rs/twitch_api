1//! Retrieves the list of available Cheermotes, animated emotes to which viewers can assign Bits, to cheer in chat.
2//! [`get-cheermotes`](https://dev.twitch.tv/docs/api/reference#get-cheermotes)
3//!
4//! # Accessing the endpoint
5//!
6//! ## Request: [GetCheermotesRequest]
7//!
8//! To use this endpoint, construct a [`GetCheermotesRequest`] with the [`GetCheermotesRequest::new()`] or [`GetCheermotesRequest::builder()`] method.
9//! If you do not provide an ID, the request will only include global cheermotes as defined by twitch.
10//!
11//! ```rust
12//! use twitch_api::helix::bits::get_cheermotes;
13//! let request = get_cheermotes::GetCheermotesRequest::broadcaster_id("1234");
14//! // Without broadcaster ID
15//! let request = get_cheermotes::GetCheermotesRequest::new();
16//! ```
17//!
18//! ## Response: [Cheermote]
19//!
20//! Send the request to receive the response with [`HelixClient::req_get()`](helix::HelixClient::req_get).
21//!
22//! ```rust, no_run
23//! use twitch_api::helix::{self, bits::get_cheermotes};
24//! # use twitch_api::client;
25//! # #[tokio::main]
26//! # async fn main() -> Result<(), Box<dyn std::error::Error + Send + Sync + 'static>> {
27//! # let client: helix::HelixClient<'static, client::DummyHttpClient> = helix::HelixClient::default();
28//! # let token = twitch_oauth2::AccessToken::new("validtoken".to_string());
29//! # let token = twitch_oauth2::UserToken::from_existing(&client, token, None, None).await?;
30//! let request = get_cheermotes::GetCheermotesRequest::new();
31//! let response: Vec<get_cheermotes::Cheermote> = client.req_get(request, &token).await?.data;
32//! # Ok(())
33//! # }
34//! ```
35//!
36//! You can also get the [`http::Request`] with [`request.create_request(&token, &client_id)`](helix::RequestGet::create_request)
37//! and parse the [`http::Response`] with [`GetCheermotesRequest::parse_response(None, &request.get_uri(), response)`](GetCheermotesRequest::parse_response)
38use super::*;
39use helix::RequestGet;

41/// Query Parameters for [Get Cheermotes](super::get_cheermotes)
42///
43/// [`get-cheermotes`](https://dev.twitch.tv/docs/api/reference#get-cheermotes)
44#[derive(PartialEq, Eq, Deserialize, Serialize, Clone, Debug, Default)]
45#[cfg_attr(feature = "typed-builder", derive(typed_builder::TypedBuilder))]
46#[must_use]
47#[non_exhaustive]
48pub struct GetCheermotesRequest<'a> {
49    /// ID for the broadcaster who might own specialized Cheermotes.
50    #[cfg_attr(feature = "typed-builder", builder(default, setter(into)))]
51    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
52    pub broadcaster_id: Option<Cow<'a, types::UserIdRef>>,
53}
54
55impl<'a> GetCheermotesRequest<'a> {
56    /// Get available Cheermotes.
57    pub fn new() -> Self { Self::default() }
58
59    /// Get Cheermotes in a specific broadcasters channel.
60    ///
61    /// # Examples
62    ///
63    /// ```rust
64    /// use twitch_api::helix::bits::GetCheermotesRequest;
65    /// let request = GetCheermotesRequest::broadcaster_id("1234");
66    /// ```
67    pub fn broadcaster_id(broadcaster_id: impl types::IntoCow<'a, types::UserIdRef> + 'a) -> Self {
68        Self {
69            broadcaster_id: Some(broadcaster_id.into_cow()),
70        }
71    }
72}

74/// Return Values for [Get Cheermotes](super::get_cheermotes)
75///
76/// [`get-cheermotes`](https://dev.twitch.tv/docs/api/reference#get-cheermotes)
77#[derive(PartialEq, Eq, Deserialize, Serialize, Debug, Clone)]
78#[cfg_attr(feature = "deny_unknown_fields", serde(deny_unknown_fields))]
79#[non_exhaustive]
80pub struct Cheermote {
81    /// Indicates whether or not this emote provides a charity contribution match during charity campaigns.
82    pub is_charitable: bool,
83    /// The data when this Cheermote was last updated.
84    pub last_updated: String,
85    /// Order of the emotes as shown in the bits card, in ascending order.
86    pub order: i64,
87    /// Prefix for cheermote
88    pub prefix: String,
89    /// An array of Cheermotes with their metadata.
90    pub tiers: Vec<Tiers>,
91    /// Shows whether the emote is `global_first_party`,  `global_third_party`, `channel_custom`, `display_only`, or `sponsored`.
92    #[serde(rename = "type")]
93    pub type_: CheermoteType,
94}
95
96/// Types of cheermotes
97#[derive(PartialEq, Eq, Deserialize, Serialize, Debug, Clone)]
98#[cfg_attr(feature = "deny_unknown_fields", serde(deny_unknown_fields))]
99#[serde(rename_all = "snake_case")]
100#[non_exhaustive]
101pub enum CheermoteType {
102    /// Global cheermote
103    GlobalFirstParty,
104    /// Global third party cheermote
105    GlobalThirdParty,
106    /// Custom channel cheermote
107    ChannelCustom,
108    /// Display only
109    ///
110    /// # Note
111    ///
112    /// Usage unknown
113    DisplayOnly,
114    /// Sponsored cheermote
115    Sponsored,
116}
117
118/// Information about tier of emote
119#[derive(PartialEq, Eq, Deserialize, Serialize, Debug, Clone)]
120#[cfg_attr(feature = "deny_unknown_fields", serde(deny_unknown_fields))]
121#[non_exhaustive]
122pub struct Tiers {
123    /// Indicates whether or not emote information is accessible to users.
124    pub can_cheer: bool,
125    /// Hex code for the color associated with the bits of that tier. Grey, Purple, Teal, Blue, or Red color to match the base bit type.
126    pub color: String,
127    /// ID of the emote tier. Possible tiers are: 1,100,500,1000,5000, 10k, or 100k.
128    pub id: String,
129    /// Structure containing both animated and static image sets, sorted by light and dark.
130    pub images: CheermoteImages,
131    /// Minimum number of bits needed to be used to hit the given tier of emote.
132    pub min_bits: i64,
133    /// Indicates whether or not we hide the emote from the bits card.
134    pub show_in_bits_card: bool,
135}

137/// Images for different themes
138#[derive(PartialEq, Eq, Deserialize, Serialize, Debug, Clone)]
139#[cfg_attr(feature = "deny_unknown_fields", serde(deny_unknown_fields))]
140#[non_exhaustive]
141pub struct CheermoteImages {

143    pub dark: CheermoteImage,
144    /// Light mode
145    pub light: CheermoteImage,

148/// URLs to the same image, animated or static.
149#[derive(PartialEq, Eq, Deserialize, Serialize, Debug, Clone)]
150#[cfg_attr(feature = "deny_unknown_fields", serde(deny_unknown_fields))]
151#[non_exhaustive]
152pub struct CheermoteImage {
153    /// Animated GIFs
154    pub animated: CheermoteImageArray,
155    /// Static PNGs
156    #[serde(rename = "static")]
157    pub static_: CheermoteImageArray,
158}
159/// Images in sizes
160#[derive(PartialEq, Eq, Deserialize, Serialize, Debug, Clone)]
161#[cfg_attr(feature = "deny_unknown_fields", serde(deny_unknown_fields))]
162#[non_exhaustive]
163pub struct CheermoteImageArray {
164    /// URL to image in size 28 × 28
165    #[serde(rename = "1")]
166    pub url_1x: String,
167    /// URL to image in size 42 × 42
168    #[serde(rename = "1.5")]
169    pub url_1_5x: String,
170    /// URL to image in size 56 × 56
171    #[serde(rename = "2")]
172    pub url_2x: String,
173    /// URL to image in size 84 × 84
174    #[serde(rename = "3")]
175    pub url_3x: String,
176    /// URL to image in size 112 × 112
177    #[serde(rename = "4")]
178    pub url_4x: String,
179}

181/// Level of cheermote image

184#[derive(PartialEq, Eq, Hash, Deserialize, Debug, Clone)]
