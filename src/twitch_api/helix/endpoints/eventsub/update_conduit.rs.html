88#[cfg(test)]
89#[test]
90fn test_successful_response() {
91    use helix::*;
92    let req = UpdateConduitRequest::default();
93
94    let body = UpdateConduitBody::new("bfcfc993-26b1-b876-44d9-afe75a379dac", 5);
95    assert_eq!(
96        std::str::from_utf8(&body.try_to_body().unwrap()).unwrap(),
97        r#"{"id":"bfcfc993-26b1-b876-44d9-afe75a379dac","shard_count":5}"#
98    );
99
100    let data = br#"{
101      "data": [
102        {
103          "id": "bfcfc993-26b1-b876-44d9-afe75a379dac",
104          "shard_count": 5
105        }
106      ]
107    }
108    "#
109    .to_vec();
110    let http_response = http::Response::builder().status(200).body(data).unwrap();
111
112    let uri = req.get_uri().unwrap();
113    let response = UpdateConduitRequest::parse_response(Some(req), &uri, http_response).unwrap();
114
115    assert_eq!(
116        response.data,
117        crate::eventsub::Conduit {
118            id: "bfcfc993-26b1-b876-44d9-afe75a379dac".into(),
119            shard_count: 5,
120        },
121    );
122
123    dbg!("{:#?}", response);
124}
