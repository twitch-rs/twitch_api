1//! Gets all scheduled broadcasts or specific scheduled broadcasts from a channel’s stream schedule. Scheduled broadcasts are defined as “stream segments” in the API.
2//! [`get-channel-stream-schedule`](https://dev.twitch.tv/docs/api/reference#get-channel-stream-schedule)
3//!
4//! ## Notes
5//!
6//! See also [`get_channel_schedule`](helix::HelixClient::get_channel_schedule)
7//!
8//! ## Request: [GetChannelStreamScheduleRequest]
9//!
10//! To use this endpoint, construct a [`GetChannelStreamScheduleRequest`] with the [`GetChannelStreamScheduleRequest::broadcaster_id()`] method.
11//!
12//! ```rust
13//! use twitch_api::helix::schedule::get_channel_stream_schedule;
14//! let request =
15//!     get_channel_stream_schedule::GetChannelStreamScheduleRequest::broadcaster_id("1234");
16//! ```
17//!
18//! ## Response: [ScheduledBroadcasts]
19//!
20//! Send the request to receive the response with [`HelixClient::req_get()`](helix::HelixClient::req_get).
21//!
22//! ```rust, no_run
23//! use twitch_api::helix::{self, schedule::get_channel_stream_schedule};
24//! # use twitch_api::client;
25//! # #[tokio::main]
26//! # async fn main() -> Result<(), Box<dyn std::error::Error + Send + Sync + 'static>> {
27//! # let client: helix::HelixClient<'static, client::DummyHttpClient> = helix::HelixClient::default();
28//! # let token = twitch_oauth2::AccessToken::new("validtoken".to_string());
29//! # let token = twitch_oauth2::UserToken::from_existing(&client, token, None, None).await?;
30//! let request = get_channel_stream_schedule::GetChannelStreamScheduleRequest::broadcaster_id("1234");
31//! let response: helix::schedule::ScheduledBroadcasts = client.req_get(request, &token).await?.data;
32//! # Ok(())
33//! # }
34//! ```
35//!
36//! You can also get the [`http::Request`] with [`request.create_request(&token, &client_id)`](helix::RequestGet::create_request)
37//! and parse the [`http::Response`] with [`GetChannelStreamScheduleRequest::parse_response(None, &request.get_uri(), response)`](GetChannelStreamScheduleRequest::parse_response)

42/// Query Parameters for [Get Channel Stream Schedule](super::get_channel_stream_schedule)
43///
44/// [`get-channel-stream-schedule`](https://dev.twitch.tv/docs/api/reference#get-channel-stream-schedule)
45#[derive(PartialEq, Eq, Deserialize, Serialize, Clone, Debug)]
46#[cfg_attr(feature = "typed-builder", derive(typed_builder::TypedBuilder))]
47#[must_use]
48#[non_exhaustive]
49pub struct GetChannelStreamScheduleRequest<'a> {
50    /// User ID of the broadcaster who owns the channel streaming schedule. Provided broadcaster_id must match the user_id in the user OAuth token.
51    #[cfg_attr(feature = "typed-builder", builder(setter(into)))]
52    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
53    pub broadcaster_id: Cow<'a, types::UserIdRef>,
54    /// The ID of the stream segment to return. Maximum: 100.
55    #[cfg_attr(feature = "typed-builder", builder(default, setter(into)))]
56    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
57    pub id: Option<Cow<'a, types::StreamSegmentIdRef>>,
58    /// A timestamp in RFC3339 format to start returning stream segments from. If not specified, the current date and time is used.
59    #[cfg_attr(feature = "typed-builder", builder(default, setter(into)))]
60    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
61    pub start_time: Option<Cow<'a, types::TimestampRef>>,
62    /// A timezone offset for the requester specified in minutes. This is recommended to ensure stream segments are returned for the correct week. For example, a timezone that is +4 hours from GMT would be “240.” If not specified, “0” is used for GMT.
63    #[cfg_attr(feature = "typed-builder", builder(default, setter(into)))]
64    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
65    pub utc_offset: Option<Cow<'a, str>>,
66    /// Cursor for forward pagination: tells the server where to start fetching the next set of results, in a multi-page response. The cursor value specified here is from the pagination response field of a prior query.
67    #[cfg_attr(feature = "typed-builder", builder(default))]
68    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
69    pub after: Option<Cow<'a, helix::CursorRef>>,
70    /// Maximum number of stream segments to return. Maximum: 25. Default: 20.
71    #[cfg_attr(feature = "typed-builder", builder(default, setter(into)))]
72    pub first: Option<usize>,
73}

92    }

94    /// Set the start_time for the request.
95    pub fn start_time(
96        mut self,

101    }

106        self
