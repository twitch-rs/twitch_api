1//! Gets clip information by clip ID (one or more), broadcaster ID (one only), or game ID (one only).
2//! [`get-clips`](https://dev.twitch.tv/docs/api/reference#get-clips)
3//!
4//! # Accessing the endpoint
5//!
6//! ## Request: [GetClipsRequest]
7//!
8//! To use this endpoint, construct a [`GetClipsRequest`] with the [`GetClipsRequest::broadcaster_id()`] method.
9//!
10//! ```rust
11//! use twitch_api::helix::clips::get_clips;
12//! let request = get_clips::GetClipsRequest::broadcaster_id("1234");
13//! ```
14//!
15//! ## Response: [Clip]
16//!
17//! Send the request to receive the response with [`HelixClient::req_get()`](helix::HelixClient::req_get).
18//!
19//! ```rust, no_run
20//! use twitch_api::helix::{self, clips::get_clips};
21//! # use twitch_api::client;
22//! # #[tokio::main]
23//! # async fn main() -> Result<(), Box<dyn std::error::Error + Send + Sync + 'static>> {
24//! # let client: helix::HelixClient<'static, client::DummyHttpClient> = helix::HelixClient::default();
25//! # let token = twitch_oauth2::AccessToken::new("validtoken".to_string());
26//! # let token = twitch_oauth2::UserToken::from_existing(&client, token, None, None).await?;
27//! let request = get_clips::GetClipsRequest::broadcaster_id("1234");
28//! let response: Vec<get_clips::Clip> = client.req_get(request, &token).await?.data;
29//! # Ok(())
30//! # }
31//! ```
32//!
33//! You can also get the [`http::Request`] with [`request.create_request(&token, &client_id)`](helix::RequestGet::create_request)
34//! and parse the [`http::Response`] with [`GetClipsRequest::parse_response(None, &request.get_uri(), response)`](GetClipsRequest::parse_response)

39/// Query Parameters for [Get Clips](super::get_clips)
40///
41/// [`get-clips`](https://dev.twitch.tv/docs/api/reference#get-clips)
42#[derive(PartialEq, Eq, Deserialize, Serialize, Clone, Debug)]
43#[cfg_attr(feature = "typed-builder", derive(typed_builder::TypedBuilder))]
44#[must_use]
45#[non_exhaustive]
46pub struct GetClipsRequest<'a> {
47    /// ID of the broadcaster for whom clips are returned. The number of clips returned is determined by the first query-string parameter (default: 20). Results are ordered by view count.
48    #[cfg_attr(feature = "typed-builder", builder(default, setter(into)))]
49    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
50    pub broadcaster_id: Option<Cow<'a, types::UserIdRef>>,
51    /// ID of the game for which clips are returned. The number of clips returned is determined by the first query-string parameter (default: 20). Results are ordered by view count.
52    #[cfg_attr(feature = "typed-builder", builder(default, setter(into)))]
53    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
54    pub game_id: Option<Cow<'a, types::CategoryIdRef>>,
55    // FIXME: add types::ClipId
56    /// ID of the clip being queried. Limit: 100.
57    #[cfg_attr(feature = "typed-builder", builder(default))]
58    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
59    // FIXME: This is essentially the same as borrow, but worse
60    #[cfg_attr(not(feature = "deser_borrow"), serde(bound(deserialize = "'de: 'a")))]
61    pub id: types::Collection<'a, types::ClipId>,
62    // one of above is needed.
63    /// Cursor for forward pagination: tells the server where to start fetching the next set of results, in a multi-page response. This applies only to queries specifying broadcaster_id or game_id. The cursor value specified here is from the pagination response field of a prior query.
64    #[cfg_attr(feature = "typed-builder", builder(default))]
65    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
66    pub after: Option<Cow<'a, helix::CursorRef>>,
67    /// Cursor for backward pagination: tells the server where to start fetching the next set of results, in a multi-page response. This applies only to queries specifying broadcaster_id or game_id. The cursor value specified here is from the pagination response field of a prior query.
68    #[cfg_attr(feature = "typed-builder", builder(default))]
69    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
70    pub before: Option<Cow<'a, helix::CursorRef>>,
71    /// Ending date/time for returned clips, in RFC3339 format. (Note that the seconds value is ignored.) If this is specified, started_at also must be specified; otherwise, the time period is ignored.
72    #[cfg_attr(feature = "typed-builder", builder(default))]
73    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
74    pub ended_at: Option<Cow<'a, types::TimestampRef>>,
75    /// Maximum number of objects to return. Maximum: 100. Default: 20.
76    #[cfg_attr(feature = "typed-builder", builder(default, setter(into)))]
77    pub first: Option<usize>,
78    /// Starting date/time for returned clips, in RFC3339 format. (Note that the seconds value is ignored.) If this is specified, ended_at also should be specified; otherwise, the ended_at date/time will be 1 week after the started_at value.
79    #[cfg_attr(feature = "typed-builder", builder(default))]
80    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
81    pub started_at: Option<Cow<'a, types::TimestampRef>>,
82    /// Determines whether the response includes featured clips. If true, returns only clips that are featured. If false, returns only clips that aren’t featured. All clips are returned if this parameter is not present.
83    #[cfg_attr(feature = "typed-builder", builder(default))]
84    pub is_featured: Option<bool>,
85}

87impl<'a> GetClipsRequest<'a> {
88    /// An empty request
89    ///
90    /// # Notes
91    ///
92    /// This is not a valid request and needs to be filled out with other fields
93    pub fn empty() -> Self {
94        Self {
95            broadcaster_id: Default::default(),
96            game_id: Default::default(),
97            id: types::Collection::default(),
98            after: Default::default(),
99            before: Default::default(),
100            ended_at: Default::default(),
101            first: Default::default(),
102            started_at: Default::default(),
103            is_featured: None,
104        }
105    }
106
107    /// Broadcaster for whom clips are returned.
108    pub fn broadcaster_id(broadcaster_id: impl types::IntoCow<'a, types::UserIdRef> + 'a) -> Self {
109        Self {
110            broadcaster_id: Some(broadcaster_id.into_cow()),
111            ..Self::empty()
112        }
113    }
114
115    /// Game for which clips are returned.
116    pub fn game_id(game_id: impl types::IntoCow<'a, types::CategoryIdRef> + 'a) -> Self {
117        Self {
118            game_id: Some(game_id.into_cow()),
119            ..Self::empty()
120        }
121    }
122
123    /// IDs of clips being queried
124    pub fn clip_ids(clip_ids: impl Into<types::Collection<'a, types::ClipId>>) -> Self {
125        Self {
126            id: clip_ids.into(),
127            ..Self::empty()
128        }
129    }
130
131    /// Ending date/time for the returned clips
132    pub fn started_at(
133        &mut self,
134        started_at: impl types::IntoCow<'a, types::TimestampRef> + 'a,
135    ) -> &mut Self {
136        self.started_at = Some(started_at.into_cow());
137        self
138    }
139
140    /// Ending date/time for the returned clips
141    pub fn ended_at(
142        &mut self,
143        ended_at: impl types::IntoCow<'a, types::TimestampRef> + 'a,
144    ) -> &mut Self {
145        self.ended_at = Some(ended_at.into_cow());
146        self
147    }
148
149    /// Set amount of results returned per page.
150    pub fn first(mut self, first: usize) -> Self {
151        self.first = Some(first);
152        self
153    }
154}

156/// Return Values for [Get Clips](super::get_clips)
157///
158/// [`get-clips`](https://dev.twitch.tv/docs/api/reference#get-clips)
159#[derive(PartialEq, Deserialize, Serialize, Debug, Clone)]
160#[cfg_attr(feature = "deny_unknown_fields", serde(deny_unknown_fields))]
161#[non_exhaustive]
162pub struct Clip {
163    /// User ID of the stream from which the clip was created.
164    pub broadcaster_id: types::UserId,
165    /// Display name corresponding to broadcaster_id.
166    pub broadcaster_name: types::DisplayName,
167    /// Date when the clip was created.
168    pub created_at: types::Timestamp,
169    /// ID of the user who created the clip.
170    pub creator_id: types::UserId,
171    /// Display name corresponding to creator_id.
172    pub creator_name: types::DisplayName,
173    /// Duration of the Clip in seconds (up to 0.1 precision).
174    pub duration: f64,
175    /// URL to embed the clip.
176    pub embed_url: String,
177    /// ID of the game assigned to the stream when the clip was created.
178    pub game_id: types::CategoryId,
179    /// ID of the clip being queried.
180    pub id: String,
181    /// Language of the stream from which the clip was created.
182    pub language: String,
183    /// URL of the clip thumbnail.
184    pub thumbnail_url: String,
185    /// Title of the clip.
186    pub title: String,
187    /// URL where the clip can be viewed.
188    pub url: String,
189    /// ID of the video from which the clip was created.
190    pub video_id: types::VideoId,
191    /// Number of times the clip has been viewed.
192    pub view_count: i64,
193    /// The zero-based offset, in seconds, to where the clip starts in the video (VOD).
194    ///
195    /// Is none if the video is not available or hasn’t been created yet from the live stream. See [video_id](Clip::video_id).
196    pub vod_offset: Option<i64>,
197    /// Indicates if the clip is featured or not.
198    pub is_featured: bool,
199}
