42/// Query Parameters for [Create Clip From VOD](super::create_clip_from_vod)
43///
44/// The duration of a clip can be from 5 seconds to 60 seconds in length, with a default of 30 seconds if not specified.
45///
46/// [`vod_offset`][CreateClipFromVodRequest::vod_offset] indicates where the clip will end.
47/// In other words, the clip will start at
48/// ([`vod_offset`][CreateClipFromVodRequest::vod_offset] - [`duration`][CreateClipFromVodRequest::duration])
49/// and end at [`vod_offset`][CreateClipFromVodRequest::vod_offset].
50/// This means that the value of [`vod_offset`][CreateClipFromVodRequest::vod_offset]
51/// must greater than or equal to the value of duration.
52///
53/// [`create-clip-from-vod`](https://dev.twitch.tv/docs/api/reference/#create-clip-from-vod)
54#[derive(PartialEq, Deserialize, Serialize, Clone, Debug)]
55#[cfg_attr(feature = "typed-builder", derive(typed_builder::TypedBuilder))]
56#[must_use]
57#[non_exhaustive]
58pub struct CreateClipFromVodRequest<'a> {
59    /// The user ID of the editor for the channel you want to create a clip for.
60    /// If using the broadcaster’s auth token, this is the same as broadcaster_id.
61    /// This must match the user_id in the user access token.
62    #[cfg_attr(feature = "typed-builder", builder(setter(into)))]
63    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
64    pub editor_id: Cow<'a, types::UserIdRef>,
65
66    /// The user ID for the channel you want to create a clip for.
67    #[cfg_attr(feature = "typed-builder", builder(setter(into)))]
68    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
69    pub broadcaster_id: Cow<'a, types::UserIdRef>,
70
71    /// ID of the VOD the user wants to clip.
72    #[cfg_attr(feature = "typed-builder", builder(setter(into)))]
73    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
74    pub vod_id: Cow<'a, types::VideoIdRef>,
75
76    /// Offset in the VOD to create the clip. See [CreateClipFromVodRequest].
77    pub vod_offset: u64,
78
79    /// The length of the clip, in seconds. Precision is 0.1. Defaults to 30. Min: 5 seconds, Max: 60 seconds.
80    #[cfg_attr(feature = "typed-builder", builder(default, setter(into)))]
81    pub duration: Option<f32>,
82
83    /// The title of the clip.
84    #[cfg_attr(feature = "typed-builder", builder(setter(into)))]
85    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
86    pub title: Cow<'a, str>,
87}

89impl<'a> CreateClipFromVodRequest<'a> {

93        broadcaster_id: impl types::IntoCow<'a, types::UserIdRef> + 'a,
94        vod_id: impl types::IntoCow<'a, types::VideoIdRef> + 'a,
95        vod_offset: u64,
96        title: impl Into<Cow<'a, str>>,
97    ) -> Self {
98        Self {
99            editor_id: editor_id.into_cow(),
100            broadcaster_id: broadcaster_id.into_cow(),
101            vod_id: vod_id.into_cow(),
102            vod_offset,
103            duration: None,
104            title: title.into(),
105        }
106    }
107
108    /// Sets the duration parameter
109    pub const fn duration(mut self, duration: f32) -> Self {
110        self.duration = Some(duration);
111        self
112    }
113}
