1//! Updates a Custom Reward created on a channel.
2//!
3//! Only rewards created programmatically by the same client_id can be updated.
4//! [`update-custom-reward`](https://dev.twitch.tv/docs/api/reference#update-custom-reward)
5//!
6//! # Accessing the endpoint
7//!
8//! ## Request: [UpdateCustomRewardRequest]
9//!
10//! To use this endpoint, construct an [`UpdateCustomRewardRequest`] with the [`UpdateCustomRewardRequest::new()`] method.
11//!
12//! ```rust
13//! use twitch_api::helix::points::update_custom_reward;
14//! let request = update_custom_reward::UpdateCustomRewardRequest::new(
15//!     "274637212",
16//!     "reward-id",
17//! );
18//! ```
19//!
20//! ## Body: [UpdateCustomRewardBody]
21//!
22//! We also need to provide a body to the request containing what we want to change.
23//!
24//! ```
25//! # use twitch_api::helix::points::update_custom_reward;
26//! let mut body = update_custom_reward::UpdateCustomRewardBody::default();
27//! body.cost = Some(501);
28//! body.title = Some("hydrate but differently now!".into());
29//! ```
30//!
31//! ## Response: [UpdateCustomReward]
32//!
33//!
34//! Send the request to receive the response with [`HelixClient::req_patch()`](helix::HelixClient::req_patch).
35//!
36//!
37//! ```rust, no_run
38//! use twitch_api::helix::{self, points::update_custom_reward};
39//! # use twitch_api::client;
40//! # #[tokio::main]
41//! # async fn main() -> Result<(), Box<dyn std::error::Error + Send + Sync + 'static>> {
42//! # let client: helix::HelixClient<'static, client::DummyHttpClient> = helix::HelixClient::default();
43//! # let token = twitch_oauth2::AccessToken::new("validtoken".to_string());
44//! # let token = twitch_oauth2::UserToken::from_existing(&client, token, None, None).await?;
45//! let request = update_custom_reward::UpdateCustomRewardRequest::new("274637212", "reward-id");
46//! let mut body = update_custom_reward::UpdateCustomRewardBody::default();
47//! body.cost = Some(501);
48//! body.title = Some("hydrate but differently now!".into());
49//! let response: update_custom_reward::UpdateCustomReward = client.req_patch(request, body, &token).await?.data;
50//! # Ok(())
51//! # }
52//! ```
53//!
54//! You can also get the [`http::Request`] with [`request.create_request(&token, &client_id)`](helix::RequestPost::create_request)
55//! and parse the [`http::Response`] with [`UpdateCustomRewardRequest::parse_response(None, &request.get_uri(), response)`](UpdateCustomRewardRequest::parse_response)

190    ) -> Result<helix::Response<Self, Self::Response>, helix::HelixRequestPatchError>
191    where

198                        HelixRequestPatchError::DeserializeError(
199                            response.to_string(),
200                            e,
201                            uri.clone(),
202                            status,
203                        )
204                    })?;
205                UpdateCustomReward::Success(resp.data.into_iter().next().ok_or(
206                    helix::HelixRequestPatchError::InvalidResponse {
207                        reason: "expected at least one element in data",
208                        response: response.to_string(),
209                        status,
210                        uri: uri.clone(),
211                    },
212                )?)
213            }
214            _ => {
215                return Err(helix::HelixRequestPatchError::InvalidResponse {
216                    reason: "unexpected status code",
217                    response: response.to_string(),
218                    status,
219                    uri: uri.clone(),
220                })
221            }
222        };
223        Ok(helix::Response::with_data(resp, request))
224    }
