1//! Updates the status of Custom Reward Redemption objects on a channel that are in the UNFULFILLED status.
2//!
3//! Only redemptions for a reward created by the same client_id as attached to the access token can be updated.
4//! [`update-redemption-status`](https://dev.twitch.tv/docs/api/reference#update-redemption-status)
5//!
6//! # Accessing the endpoint
7//!
8//! ## Request: [UpdateRedemptionStatusRequest]
9//!
10//! To use this endpoint, construct a [`UpdateRedemptionStatusRequest`] with the [`UpdateRedemptionStatusRequest::new()`] method.
11//!
12//! ```rust
13//! use twitch_api::helix::points::UpdateRedemptionStatusRequest;
14//! let request = UpdateRedemptionStatusRequest::new(
15//!     "274637212",
16//!     "92af127c-7326-4483-a52b-b0da0be61c01",
17//!     "17fa2df1-ad76-4804-bfa5-a40ef63efe63",
18//! );
19//! ```
20//!
21//! ## Body: [UpdateRedemptionStatusBody]
22//!
23//! We also need to provide a body to the request containing what we want to change.
24//!
25//! ```
26//! use twitch_api::helix::points::{
27//!     CustomRewardRedemptionStatus, UpdateRedemptionStatusBody,
28//! };
29//! let body = UpdateRedemptionStatusBody::status(
30//!     CustomRewardRedemptionStatus::Canceled,
31//! );
32//! ```
33//!
34//! ## Response: [CustomRewardRedemption]
35//!
36//! Send the request to receive the response with [`HelixClient::req_get()`](helix::HelixClient::req_get).
37//!
38//! ```rust, no_run
39//! use twitch_api::helix;
40//! use twitch_api::helix::points::{
41//!     CustomRewardRedemptionStatus, UpdateRedemptionStatusBody,
42//!     CustomRewardRedemption, UpdateRedemptionStatusRequest,
43//! };
44//! # use twitch_api::client;
45//! # #[tokio::main]
46//! # async fn main() -> Result<(), Box<dyn std::error::Error + Send + Sync + 'static>> {
47//! let client: helix::HelixClient<'static, client::DummyHttpClient> =
48//!     helix::HelixClient::default();
49//! # let token = twitch_oauth2::AccessToken::new("validtoken".to_string());
50//! # let token = twitch_oauth2::UserToken::from_existing(&client, token, None, None).await?;
51//! let request = UpdateRedemptionStatusRequest::new(
52//!     "274637212",
53//!     "92af127c-7326-4483-a52b-b0da0be61c01",
54//!     "17fa2df1-ad76-4804-bfa5-a40ef63efe63",
55//! );
56//! let body = UpdateRedemptionStatusBody::status(CustomRewardRedemptionStatus::Canceled);
57//! let response: CustomRewardRedemption =
58//!     client.req_patch(request, body, &token).await?.data;
59//! # Ok(())
60//! # }
61//! ```
62//!
63//! You can also get the [`http::Request`] with [`request.create_request(body, &token, &client_id)`](helix::RequestPatch::create_request)
64//! and parse the [`http::Response`] with [`UpdateRedemptionStatusRequest::parse_response(None, &request.get_uri(), response)`](UpdateRedemptionStatusRequest::parse_response)

70/// Query Parameters for [Update Redemption Status](super::update_redemption_status)
71///
72/// [`update-redemption-status`](https://dev.twitch.tv/docs/api/reference#update-redemption-status)
73#[derive(PartialEq, Eq, Deserialize, Serialize, Clone, Debug)]
74#[cfg_attr(feature = "typed-builder", derive(typed_builder::TypedBuilder))]
75#[must_use]
76#[non_exhaustive]
77pub struct UpdateRedemptionStatusRequest<'a> {
78    /// Provided broadcaster_id must match the user_id in the auth token.
79    #[cfg_attr(feature = "typed-builder", builder(setter(into)))]
80    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
81    pub broadcaster_id: Cow<'a, types::UserIdRef>,
82
83    /// ID of the Custom Reward the redemptions to be updated are for.
84    #[cfg_attr(feature = "typed-builder", builder(setter(into)))]
85    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
86    pub reward_id: Cow<'a, types::RewardIdRef>,
87
88    /// ID of the Custom Reward Redemption to update, must match a Custom Reward Redemption on broadcaster_id’s channel
89    #[cfg_attr(feature = "typed-builder", builder(setter(into)))]
90    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
91    pub id: Cow<'a, types::RedemptionIdRef>,
92}
