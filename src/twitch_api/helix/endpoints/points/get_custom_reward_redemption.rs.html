1//! Returns Custom Reward Redemption objects for a Custom Reward on a channel that was created by the same client_id.
2//!
3//! Developers only have access to get and update redemptions for the rewards they created.
4//! [`get-custom-reward-redemption`](https://dev.twitch.tv/docs/api/reference#get-custom-reward-redemption)
5//!
6//! # Accessing the endpoint
7//!
8//! ## Request: [GetCustomRewardRedemptionRequest]
9//!
10//! To use this endpoint, construct a [`GetCustomRewardRedemptionRequest`] with the [`GetCustomRewardRedemptionRequest::broadcaster_id()`] method.
11//!
12//! ```rust
13//! use twitch_api::helix::points::{
14//!     CustomRewardRedemptionStatus, GetCustomRewardRedemptionRequest,
15//! };
16//! let request = GetCustomRewardRedemptionRequest::broadcaster_id("274637212")
17//!     .reward_id("92af127c-7326-4483-a52b-b0da0be61c01")
18//!     .status(CustomRewardRedemptionStatus::Canceled);
19//! ```
20//!
21//! ## Response: [CustomRewardRedemption]
22//!
23//! Send the request to receive the response with [`HelixClient::req_get()`](helix::HelixClient::req_get).
24//!
25//! ```rust, no_run
26//! use twitch_api::helix;
27//! use twitch_api::helix::points::{CustomRewardRedemptionStatus, CustomRewardRedemption, GetCustomRewardRedemptionRequest};
28//! # use twitch_api::client;
29//! # #[tokio::main]
30//! # async fn main() -> Result<(), Box<dyn std::error::Error + Send + Sync + 'static>> {
31//! # let client: helix::HelixClient<'static, client::DummyHttpClient> = helix::HelixClient::default();
32//! # let token = twitch_oauth2::AccessToken::new("validtoken".to_string());
33//! # let token = twitch_oauth2::UserToken::from_existing(&client, token, None, None).await?;
34//! let request = GetCustomRewardRedemptionRequest::broadcaster_id("274637212")
35//!     .reward_id("92af127c-7326-4483-a52b-b0da0be61c01")
36//!     .status(CustomRewardRedemptionStatus::Canceled);
37//! let response: Vec<CustomRewardRedemption> = client.req_get(request, &token).await?.data;
38//! # Ok(())
39//! # }
40//! ```
41//!
42//! You can also get the [`http::Request`] with [`request.create_request(&token, &client_id)`](helix::RequestGet::create_request)
43//! and parse the [`http::Response`] with [`GetCustomRewardRedemptionRequest::parse_response(None, &request.get_uri(), response)`](GetCustomRewardRedemptionRequest::parse_response)

48/// Query Parameters for [Get Custom Reward Redemption](super::get_custom_reward_redemption)
49///
50/// [`get-custom-reward-redemption`](https://dev.twitch.tv/docs/api/reference#get-custom-reward-redemption)
51#[derive(PartialEq, Eq, Deserialize, Serialize, Clone, Debug)]
52#[cfg_attr(feature = "typed-builder", derive(typed_builder::TypedBuilder))]
53#[must_use]
54#[non_exhaustive]
55pub struct GetCustomRewardRedemptionRequest<'a> {
56    /// Provided broadcaster_id must match the user_id in the auth token
57    #[cfg_attr(feature = "typed-builder", builder(setter(into)))]
58    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
59    pub broadcaster_id: Cow<'a, types::UserIdRef>,
60
61    /// When ID is not provided, this parameter returns paginated Custom Reward Redemption objects for redemptions of the Custom Reward with ID reward_id
62    #[cfg_attr(feature = "typed-builder", builder(setter(into)))]
63    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
64    pub reward_id: Option<Cow<'a, types::RewardIdRef>>,
65
66    /// When id is not provided, this param is required and filters the paginated Custom Reward Redemption objects for redemptions with the matching status. Can be one of UNFULFILLED, FULFILLED or CANCELED
67    #[cfg_attr(feature = "typed-builder", builder(default, setter(into)))]
68    pub status: Option<CustomRewardRedemptionStatus>,
69
70    /// A list of IDs to filter the redemptions by.
71    #[cfg_attr(feature = "typed-builder", builder(setter(into), default))]
72    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
73    // FIXME: This is essentially the same as borrow, but worse
74    #[cfg_attr(not(feature = "deser_borrow"), serde(bound(deserialize = "'de: 'a")))]
75    pub id: types::Collection<'a, types::RedemptionId>,
76
77    /// The order to sort redemptions by.
78    #[cfg_attr(feature = "typed-builder", builder(default))]
79    pub sort: Option<GetCustomRewardRedemptionSortOrder>,
80
81    /// Cursor for forward pagination: tells the server where to start fetching the next set of results, in a multi-page response. This applies only to queries without ID. If an ID is specified, it supersedes any cursor/offset combinations. The cursor value specified here is from the pagination response field of a prior query.
82    #[cfg_attr(feature = "typed-builder", builder(default))]
83    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
84    pub after: Option<Cow<'a, helix::CursorRef>>,
85
86    /// Number of results to be returned when getting the paginated Custom Reward Redemption objects for a reward. Limit: 50. Default: 20.
87    #[cfg_attr(feature = "typed-builder", builder(default, setter(into)))]
88    pub first: Option<usize>,
89}

91impl<'a> GetCustomRewardRedemptionRequest<'a> {
92    /// Reward to fetch
93    pub fn broadcaster_id(broadcaster_id: impl types::IntoCow<'a, types::UserIdRef> + 'a) -> Self {
94        Self {
95            broadcaster_id: broadcaster_id.into_cow(),
96            reward_id: None,
97            status: Default::default(),
98            after: Default::default(),
99            first: Default::default(),
100            id: Default::default(),
101            sort: None,
102        }
103    }
104
105    /// Specific reward to query
106    pub fn reward_id(
107        mut self,
108        reward_id: impl types::IntoCow<'a, types::RewardIdRef> + 'a,
109    ) -> Self {
110        self.reward_id = Some(reward_id.into_cow());
111        self
112    }
113
114    /// Status of rewards to return
115    pub fn status(mut self, status: impl Into<CustomRewardRedemptionStatus>) -> Self {
116        self.status = Some(status.into());
117        self
118    }
119
120    /// The order to sort redemptions by.
121    pub fn sort(mut self, sort: impl Into<GetCustomRewardRedemptionSortOrder>) -> Self {
122        self.sort = Some(sort.into());
123        self
124    }
125
126    /// Get redemptions with these ids
127    pub fn ids(mut self, id: impl Into<types::Collection<'a, types::RedemptionId>>) -> Self {
128        self.id = id.into();
129        self
130    }
131}

144/// Return Values for [Get Custom Reward Redemption](super::get_custom_reward_redemption)
145///
146/// [`get-custom-reward-redemption`](https://dev.twitch.tv/docs/api/reference#get-custom-reward-redemption)
147#[derive(PartialEq, Eq, Deserialize, Serialize, Debug, Clone)]
148#[cfg_attr(feature = "deny_unknown_fields", serde(deny_unknown_fields))]
149#[non_exhaustive]
150pub struct CustomRewardRedemption {
151    /// The id of the broadcaster that the reward belongs to.
152    pub broadcaster_id: types::UserId,
153
154    /// The display name of the broadcaster that the reward belongs to.
155    pub broadcaster_name: types::DisplayName,
156
157    /// Broadcaster’s user login name.
158    pub broadcaster_login: types::UserName,
159
160    /// The ID of the redemption.
161    pub id: types::RedemptionId,
162
163    /// The ID of the user that redeemed the reward
164    pub user_id: types::UserId,
165
166    /// The display name of the user that redeemed the reward.
167    pub user_name: types::DisplayName,
168
169    ///The login of the user who redeemed the reward.
170    pub user_login: types::UserName,
171
172    /// Basic information about the Custom Reward that was redeemed at the time it was redeemed. { “id”: string, “title”: string, “prompt”: string, “cost”: int, }
173    pub reward: Reward,
174
175    /// The user input provided. Empty string if not provided.
176    pub user_input: String,
177
178    /// One of UNFULFILLED, FULFILLED or CANCELED
179    pub status: CustomRewardRedemptionStatus,
180
181    /// RFC3339 timestamp of when the reward was redeemed.
182    pub redeemed_at: types::Timestamp,
183}
