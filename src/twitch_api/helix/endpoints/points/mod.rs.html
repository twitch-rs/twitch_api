72/// Custom reward redemption statuses: UNFULFILLED, FULFILLED or CANCELED
73#[derive(PartialEq, Eq, Serialize, Deserialize, Copy, Clone, Debug)]
74#[non_exhaustive]
75pub enum CustomRewardRedemptionStatus {
76    /// Unfulfilled reward - the user has claimed it but it is still pending.
77    #[serde(rename = "UNFULFILLED")]
78    Unfulfilled,
79    /// Fulfilled reward - the user has claimed it and the reward has been granted.
80    #[serde(rename = "FULFILLED")]
81    Fulfilled,
82    /// Cancelled reward - the reward has been cancelled before fulfillment, and any spent points have been refunded.
83    #[serde(rename = "CANCELED")]
84    Canceled,
85}
