1//! Returns a list of Custom Reward objects for the Custom Rewards on a channel.
2//!
3//! Developers only have access to update and delete rewards that were created programmatically by the same/calling client_id.
4//! [`get-custom-reward-redemption`](https://dev.twitch.tv/docs/api/reference#get-custom-reward-redemption)
5//!
6//! # Accessing the endpoint
7//!
8//! ## Request: [GetCustomRewardRequest]
9//!
10//! To use this endpoint, construct a [`GetCustomRewardRequest`] with the [`GetCustomRewardRequest::broadcaster_id()`] method.
11//!
12//! ```rust
13//! use twitch_api::helix::points::GetCustomRewardRequest;
14//! let request = GetCustomRewardRequest::broadcaster_id("274637212");
15//! ```
16//!
17//! ## Response: [CustomReward]
18//!
19//! Send the request to receive the response with [`HelixClient::req_get()`](helix::HelixClient::req_get).
20//!
21//! ```rust, no_run
22//! use twitch_api::helix;
23//! use twitch_api::helix::points::{CustomReward, GetCustomRewardRequest};
24//! # use twitch_api::client;
25//! # #[tokio::main]
26//! # async fn main() -> Result<(), Box<dyn std::error::Error + Send + Sync + 'static>> {
27//! # let client: helix::HelixClient<'static, client::DummyHttpClient> = helix::HelixClient::default();
28//! # let token = twitch_oauth2::AccessToken::new("validtoken".to_string());
29//! # let token = twitch_oauth2::UserToken::from_existing(&client, token, None, None).await?;
30//! let request = GetCustomRewardRequest::broadcaster_id("274637212");
31//! let response: Vec<CustomReward> = client.req_get(request, &token).await?.data;
32//! # Ok(())
33//! # }
34//! ```
35//!
36//! You can also get the [`http::Request`] with [`request.create_request(&token, &client_id)`](helix::RequestGet::create_request)
37//! and parse the [`http::Response`] with [`GetCustomRewardRequest::parse_response(None, &request.get_uri(), response)`](GetCustomRewardRequest::parse_response)

42/// Query Parameters for [Get Custom Reward](super::get_custom_reward)
43///
44/// [`get-custom-reward`](https://dev.twitch.tv/docs/api/reference#get-custom-reward)
45#[derive(PartialEq, Eq, Deserialize, Serialize, Clone, Debug)]
46#[cfg_attr(feature = "typed-builder", derive(typed_builder::TypedBuilder))]
47#[must_use]
48#[non_exhaustive]
49pub struct GetCustomRewardRequest<'a> {
50    /// Provided broadcaster_id must match the user_id in the auth token
51    #[cfg_attr(feature = "typed-builder", builder(setter(into)))]
52    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
53    pub broadcaster_id: Cow<'a, types::UserIdRef>,
54    /// When used, this parameter filters the results and only returns reward objects for the Custom Rewards with matching ID. Maximum: 50
55    #[cfg_attr(feature = "typed-builder", builder(default, setter(into)))]
56    #[cfg_attr(feature = "deser_borrow", serde(borrow = "'a"))]
57    // FIXME: This is essentially the same as borrow, but worse
58    #[cfg_attr(not(feature = "deser_borrow"), serde(bound(deserialize = "'de: 'a")))]
59    pub id: types::Collection<'a, types::RewardId>,
60    /// When set to true, only returns custom rewards that the calling client_id can manage. Defaults false.
61    #[cfg_attr(feature = "typed-builder", builder(default))]
62    pub only_manageable_rewards: Option<bool>,
63}
64
65impl<'a> GetCustomRewardRequest<'a> {
66    /// Rewards on this broadcasters channel
67    pub fn broadcaster_id(broadcaster_id: impl types::IntoCow<'a, types::UserIdRef> + 'a) -> Self {
68        Self {
69            broadcaster_id: broadcaster_id.into_cow(),
70            id: types::Collection::default(),
71            only_manageable_rewards: Default::default(),
72        }
73    }
74
75    /// Get rewards with these ids. Maximum 50
76    pub fn ids(mut self, id: impl Into<types::Collection<'a, types::RewardId>>) -> Self {
77        self.id = id.into();
78        self
79    }
80
81    /// Only return rewards that this application/client id can edit
82    pub fn only_manageable_rewards(mut self, only_manageable_rewards: bool) -> Self {
83        self.only_manageable_rewards = Some(only_manageable_rewards);
84        self
85    }
86}

88/// Return Values for [Get Custom Reward](super::get_custom_reward)
89///
90/// [`get-custom-reward`](https://dev.twitch.tv/docs/api/reference#get-custom-reward-redemption)
91#[derive(PartialEq, Eq, Deserialize, Serialize, Debug, Clone)]
92#[cfg_attr(feature = "deny_unknown_fields", serde(deny_unknown_fields))]
93#[non_exhaustive]
94pub struct CustomReward {
95    /// ID of the channel the reward is for
96    pub broadcaster_id: types::UserId,
97    /// Login of the channel the reward is for
98    pub broadcaster_login: types::UserName,
99    /// Display name of the channel the reward is for
100    pub broadcaster_name: types::DisplayName,
101    /// ID of the reward
102    pub id: types::RewardId,
103    /// The title of the reward
104    pub title: String,
105    /// The prompt for the viewer when they are redeeming the reward
106    pub prompt: String,
107    /// The cost of the reward
108    pub cost: usize,
109    /// Set of custom images of 1x, 2x and 4x sizes for the reward { url_1x: string, url_2x: string, url_4x: string }, can be null if no images have been uploaded
110    pub image: Option<types::Image>,
111    /// Set of default images of 1x, 2x and 4x sizes for the reward { url_1x: string, url_2x: string, url_4x: string }
112    pub default_image: Option<types::Image>,
113    /// Custom background color for the reward. Format: Hex with # prefix. Example: #00E5CB.
114    pub background_color: String,
115    /// Is the reward currently enabled, if false the reward won’t show up to viewers
116    pub is_enabled: bool,
117    /// Does the user need to enter information when redeeming the reward
118    pub is_user_input_required: bool,
119    /// Whether a maximum per stream is enabled and what the maximum is. { is_enabled: bool, max_per_stream: int }
120    pub max_per_stream_setting: types::Max,
121    /// Whether a maximum per user per stream is enabled and what the maximum is. { is_enabled: bool, max_per_user_per_stream: int }
122    pub max_per_user_per_stream_setting: types::Max,
123    /// Whether a cooldown is enabled and what the cooldown is. { is_enabled: bool, global_cooldown_seconds: int }
124    pub global_cooldown_setting: types::GlobalCooldown,
125    /// Is the reward currently paused, if true viewers can’t redeem
126    pub is_paused: bool,
127    /// Is the reward currently in stock, if false viewers can’t redeem
128    pub is_in_stock: bool,
129    /// Should redemptions be set to FULFILLED status immediately when redeemed and skip the request queue instead of the normal UNFULFILLED status.
130    pub should_redemptions_skip_request_queue: bool,
131    /// The number of redemptions redeemed during the current live stream. Counts against the max_per_stream_setting limit. Null if the broadcasters stream isn’t live or max_per_stream_setting isn’t enabled.
132    pub redemptions_redeemed_current_stream: Option<usize>,
133    /// Timestamp of the cooldown expiration. Null if the reward isn’t on cooldown.
134    pub cooldown_expires_at: Option<types::Timestamp>,
135}

146}
